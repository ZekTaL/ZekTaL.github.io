181  private static void SetPositionVisibleinSceneView(RectTransform canvasRTransform, RectTransform itemTransform)
182  {
183  // Find the best scene view
184  SceneView sceneView = SceneView.lastActiveSceneView;
185 
186  if (sceneView == null && SceneView.sceneViews.Count > 0)
187  sceneView = SceneView.sceneViews[0] as SceneView;
188 
189  // Couldn't find a SceneView. Don't set position.
190  if (sceneView == null || sceneView.camera == null)
191  return;
192 
193  // Create world space Plane from canvas position.
194  Camera camera = sceneView.camera;
195  Vector3 position = Vector3.zero;
196  Vector2 localPlanePosition;
197 
198  if (RectTransformUtility.ScreenPointToLocalPointInRectangle(canvasRTransform, new Vector2(camera.pixelWidth / 2, camera.pixelHeight / 2), camera, out localPlanePosition))
199  {
200  // Adjust for canvas pivot
201  localPlanePosition.x = localPlanePosition.x + canvasRTransform.sizeDelta.x * canvasRTransform.pivot.x;
202  localPlanePosition.y = localPlanePosition.y + canvasRTransform.sizeDelta.y * canvasRTransform.pivot.y;
203 
204  localPlanePosition.x = Mathf.Clamp(localPlanePosition.x, 0, canvasRTransform.sizeDelta.x);
205  localPlanePosition.y = Mathf.Clamp(localPlanePosition.y, 0, canvasRTransform.sizeDelta.y);
206 
207  // Adjust for anchoring
208  position.x = localPlanePosition.x - canvasRTransform.sizeDelta.x * itemTransform.anchorMin.x;
209  position.y = localPlanePosition.y - canvasRTransform.sizeDelta.y * itemTransform.anchorMin.y;
210 
211  Vector3 minLocalPosition;
212  minLocalPosition.x = canvasRTransform.sizeDelta.x * (0 - canvasRTransform.pivot.x) + itemTransform.sizeDelta.x * itemTransform.pivot.x;
213  minLocalPosition.y = canvasRTransform.sizeDelta.y * (0 - canvasRTransform.pivot.y) + itemTransform.sizeDelta.y * itemTransform.pivot.y;
214 
215  Vector3 maxLocalPosition;
216  maxLocalPosition.x = canvasRTransform.sizeDelta.x * (1 - canvasRTransform.pivot.x) - itemTransform.sizeDelta.x * itemTransform.pivot.x;
217  maxLocalPosition.y = canvasRTransform.sizeDelta.y * (1 - canvasRTransform.pivot.y) - itemTransform.sizeDelta.y * itemTransform.pivot.y;
218 
219  position.x = Mathf.Clamp(position.x, minLocalPosition.x, maxLocalPosition.x);
220  position.y = Mathf.Clamp(position.y, minLocalPosition.y, maxLocalPosition.y);
221  }
222 
223  itemTransform.anchoredPosition = position;
224  itemTransform.localRotation = Quaternion.identity;
225  itemTransform.localScale = Vector3.one;
226  }

230  {

240  PrefabStage prefabStage = PrefabStageUtility.GetCurrentPrefabStage();
241  if (prefabStage != null && !prefabStage.IsPartOfPrefabContents(parent))
242  parent = prefabStage.prefabContentsRoot;

361  // Helper function that returns a Canvas GameObject; preferably a parent of the selection, or other existing Canvas.
362  public static GameObject GetOrCreateCanvasGameObject()
363  {
364  GameObject selectedGo = Selection.activeGameObject;
365 
366  // Try to find a gameobject that is the selected GO or one if its parents.
367  Canvas canvas = (selectedGo != null) ? selectedGo.GetComponentInParent<Canvas>() : null;
368  if (IsValidCanvas(canvas))
369  return canvas.gameObject;
370 
371  // No canvas in selection or its parents? Then use any valid canvas.
372  // We have to find all loaded Canvases, not just the ones in main scenes.
373  Canvas[] canvasArray = StageUtility.GetCurrentStageHandle().FindComponentsOfType<Canvas>();
374  for (int i = 0; i < canvasArray.Length; i++)
375  if (IsValidCanvas(canvasArray[i]))
376  return canvasArray[i].gameObject;
377 
378  // No canvas in the scene at all? Then create a new one.
379  return CreateNewUI();
380  }

382  static bool IsValidCanvas(Canvas canvas)
383  {
384  if (canvas == null || !canvas.gameObject.activeInHierarchy)
385  return false;
386 
387  // It's important that the non-editable canvas from a prefab scene won't be rejected,
388  // but canvases not visible in the Hierarchy at all do. Don't check for HideAndDontSave.
389  if (EditorUtility.IsPersistent(canvas) || (canvas.hideFlags & HideFlags.HideInHierarchy) != 0)
390  return false;
391 
392  if (StageUtility.GetStageHandle(canvas.gameObject) != StageUtility.GetCurrentStageHandle())
393  return false;
394 
395  return true;
396  }
