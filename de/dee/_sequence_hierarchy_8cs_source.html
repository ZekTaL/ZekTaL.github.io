227  static bool ValidateSubElements(List<SequencePathSubElement> subElements, PlayableDirector director, out List<SequenceBuildingBlock> buildingBlocks)
228  {
229  buildingBlocks = new List<SequenceBuildingBlock>(subElements.Count);
230  var currentDirector = director;
231 
232  foreach (var element in subElements)
233  {
234  var timeline = currentDirector.playableAsset as TimelineAsset;
235  if (timeline == null)
236  return false;
237  if (timeline.trackObjects == null)
238  return false;
239 
240  var track = timeline.GetOutputTracks().FirstOrDefault(t => t.GetInstanceID() == element.trackInstanceID);
241  if (track == null)
242  return false;
243  if (track.Hash() != element.trackHash)
244  return false;
245  if (track.clips == null)
246  return false;
247  if (track.clips.Length <= element.clipIndex)
248  return false;
249 
250  var clip = track.clips[element.clipIndex];
251  if (clip == null)
252  return false;
253  if (clip.Hash() != element.clipHash)
254  return false;
255 
256  var candidateDirectors = TimelineUtility.GetSubTimelines(clip, director);
257 
258  if (element.subDirectorIndex < 0 || element.subDirectorIndex >= candidateDirectors.Count)
259  return false;
260 
261  var candidateDirector = candidateDirectors[element.subDirectorIndex];
262 
263  if (candidateDirector == null || !(candidateDirector.playableAsset is TimelineAsset))
264  return false;
265 
266  currentDirector = candidateDirector;
267 
268  buildingBlocks.Add(
269  new SequenceBuildingBlock
270  {
271  asset = currentDirector.playableAsset as TimelineAsset,
272  director = currentDirector,
273  hostClip = clip
274  });
275  }
276 
277  return true;
