541  private void UpgradeSpriteAsset()
542  {
543  m_Version = "1.1.0";
544 
545  Debug.Log("Upgrading sprite asset [" + this.name + "] to version " + m_Version + ".", this);

547  // Convert legacy glyph and character tables to new format
548  m_SpriteCharacterTable.Clear();
549  m_SpriteGlyphTable.Clear();
550 
551  for (int i = 0; i < spriteInfoList.Count; i++)
552  {
553  TMP_Sprite oldSprite = spriteInfoList[i];
554 
555  TMP_SpriteGlyph spriteGlyph = new TMP_SpriteGlyph();
556  spriteGlyph.index = (uint)i;
557  spriteGlyph.sprite = oldSprite.sprite;
558  spriteGlyph.metrics = new GlyphMetrics(oldSprite.width, oldSprite.height, oldSprite.xOffset, oldSprite.yOffset, oldSprite.xAdvance);
559  spriteGlyph.glyphRect = new GlyphRect((int)oldSprite.x, (int)oldSprite.y, (int)oldSprite.width, (int)oldSprite.height);
560 
561  spriteGlyph.scale = 1.0f;
562  spriteGlyph.atlasIndex = 0;
563 
564  m_SpriteGlyphTable.Add(spriteGlyph);
565 
566  TMP_SpriteCharacter spriteCharacter = new TMP_SpriteCharacter();
567  spriteCharacter.glyph = spriteGlyph;
568  spriteCharacter.unicode = oldSprite.unicode == 0x0 ? 0xFFFE : (uint)oldSprite.unicode;
569  spriteCharacter.name = oldSprite.name;
570  spriteCharacter.scale = oldSprite.scale;
571 
572  m_SpriteCharacterTable.Add(spriteCharacter);
573  }
