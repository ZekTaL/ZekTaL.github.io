62  public static PlacementValidity GetPlacementValidity(ITimelineItem item, IEnumerable<ITimelineItem> otherItems)
63  {
64  if (item.duration <= 0.0)
65  return PlacementValidity.Valid; //items without any duration can always be placed
66 
67  var sortedItems = otherItems.Where(i => i.duration > 0.0).OrderBy(c => c.start);
68  var candidates = new List<ITimelineItem>();
69  foreach (var sortedItem in sortedItems)
70  {
71  if ((DiscreteTime)sortedItem.start >= (DiscreteTime)item.end)
72  {
73  // No need to process further
74  break;
75  }
76 
77  if ((DiscreteTime)sortedItem.end <= (DiscreteTime)item.start)
78  {
79  // Skip
80  continue;
81  }
82 
83  candidates.Add(sortedItem);
84  }
85 
86  var discreteStart = (DiscreteTime)item.start;
87  var discreteEnd = (DiscreteTime)item.end;
88 
89  // Note: Order of tests matters
90  for (int i = 0, n = candidates.Count; i < n; i++)
91  {
92  var candidate = candidates[i];
93 
94  var blendItem = item as IBlendable;
95  if (blendItem != null && blendItem.supportsBlending)
96  {
97  if (EditModeUtils.Contains(candidate.start, candidate.end, item))
98  return PlacementValidity.InvalidIsWithin;
99 
100  if (i < n - 1)
101  {
102  var nextCandidate = candidates[i + 1];
103 
104  var discreteNextCandidateStart = (DiscreteTime)nextCandidate.start;
105  var discreteCandidateEnd = (DiscreteTime)candidate.end;
106 
107  if (discreteCandidateEnd > discreteNextCandidateStart)
108  {
109  if (discreteStart >= discreteNextCandidateStart)
110  {
111  // Note: In case the placement is fully within a blend,
112  // InvalidStartsInBlend MUST have priority
113  return PlacementValidity.InvalidStartsInBlend;
114  }
115 
116  if (discreteEnd > discreteNextCandidateStart && discreteEnd <= discreteCandidateEnd)
117  return PlacementValidity.InvalidEndsInBlend;
118 
119  if (discreteStart < discreteNextCandidateStart && discreteEnd > discreteCandidateEnd)
120  return PlacementValidity.InvalidContainsBlend;
121  }
122  }
123 
124  if (EditModeUtils.Contains(item.start, item.end, candidate))
125  return PlacementValidity.InvalidContains;
126  }
127  else
128  {
129  if (EditModeUtils.Overlaps(item, candidate.start, candidate.end)
130  || EditModeUtils.Overlaps(candidate, item.start, item.end))
131  return PlacementValidity.InvalidOverlapWithNonBlendableClip;
132  }
133  }
134 
135  return PlacementValidity.Valid;
136  }
