301  protected virtual void HandleScaleWithScreenSize()
302  {
303  Vector2 screenSize = m_Canvas.renderingDisplaySize;
304 
305  // Multiple display support only when not the main display. For display 0 the reported
306  // resolution is always the desktops resolution since its part of the display API,
307  // so we use the standard none multiple display method. (case 741751)
308  int displayIndex = m_Canvas.targetDisplay;
309  if (displayIndex > 0 && displayIndex < Display.displays.Length)
310  {
311  Display disp = Display.displays[displayIndex];
312  screenSize = new Vector2(disp.renderingWidth, disp.renderingHeight);
313  }
314 
315 
316  float scaleFactor = 0;
317  switch (m_ScreenMatchMode)
318  {
319  case ScreenMatchMode.MatchWidthOrHeight:
320  {
321  // We take the log of the relative width and height before taking the average.
322  // Then we transform it back in the original space.
323  // the reason to transform in and out of logarithmic space is to have better behavior.
324  // If one axis has twice resolution and the other has half, it should even out if widthOrHeight value is at 0.5.
325  // In normal space the average would be (0.5 + 2) / 2 = 1.25
326  // In logarithmic space the average is (-1 + 1) / 2 = 0
327  float logWidth = Mathf.Log(screenSize.x / m_ReferenceResolution.x, kLogBase);
328  float logHeight = Mathf.Log(screenSize.y / m_ReferenceResolution.y, kLogBase);
329  float logWeightedAverage = Mathf.Lerp(logWidth, logHeight, m_MatchWidthOrHeight);
330  scaleFactor = Mathf.Pow(kLogBase, logWeightedAverage);
331  break;
332  }
333  case ScreenMatchMode.Expand:
334  {
335  scaleFactor = Mathf.Min(screenSize.x / m_ReferenceResolution.x, screenSize.y / m_ReferenceResolution.y);
336  break;
337  }
338  case ScreenMatchMode.Shrink:
339  {
340  scaleFactor = Mathf.Max(screenSize.x / m_ReferenceResolution.x, screenSize.y / m_ReferenceResolution.y);
341  break;
342  }
343  }
344 
345  SetScaleFactor(scaleFactor);
346  SetReferencePixelsPerUnit(m_ReferencePixelsPerUnit);
347  }
