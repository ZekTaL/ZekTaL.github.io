726  static double GetExtrapolatedTime(double time, ClipExtrapolation mode, double duration)
727  {
728  if (duration == 0)
729  return 0;
730 
731  switch (mode)
732  {
733  case ClipExtrapolation.None:
734  break;
735 
736  case ClipExtrapolation.Loop:
737  if (time < 0)
738  time = duration - (-time % duration);
739  else if (time > duration)
740  time %= duration;
741  break;
742 
743  case ClipExtrapolation.Hold:
744  if (time < 0)
745  return 0;
746  if (time > duration)
747  return duration;
748  break;
749 
750  case ClipExtrapolation.PingPong:
751  if (time < 0)
752  {
753  time = duration * 2 - (-time % (duration * 2));
754  time = duration - Math.Abs(time - duration);
755  }
756  else
757  {
758  time = time % (duration * 2.0);
759  time = duration - Math.Abs(time - duration);
760  }
761  break;
762 
763  case ClipExtrapolation.Continue:
764  break;
765  }
766  return time;
767  }
