157  for (int i = 0; i < m_InternalUpdateQueue.Count; i++)

162  // Handle Layout Rebuild Phase

167 
168  if (m_LayoutRebuildQueue.Count > 0)

172  }
173 
174  // Handle Graphic Rebuild Phase
175  for (int i = 0; i < m_GraphicRebuildQueue.Count; i++)
176  {
177  m_GraphicRebuildQueue[i].Rebuild(CanvasUpdate.PreRender);
178  }
179 
180  // If there are no objects in the queue, we don't need to clear the lists again.
181  if (m_GraphicRebuildQueue.Count > 0)
182  {
183  m_GraphicRebuildQueue.Clear();
184  m_GraphicQueueLookup.Clear();
185  }
186 
187  // Handle Culling Update

189  {
