316  static ProjectStatus GetNewProjectStatus(CollabInfo collabInfo, ConnectInfo connectInfo, ProjectInfo projectInfo)
317  {
318  // No UPID.
319  if (!projectInfo.projectBound)
320  {
321  return ProjectStatus.Unbound;
322  }
323 
324  if (!connectInfo.online)
325  {
326  return ProjectStatus.Offline;
327  }
328 
329  if (connectInfo.maintenance || collabInfo.maintenance)
330  {
331  return ProjectStatus.Maintenance;
332  }

336  return ProjectStatus.LoggedOut;

346  {
347  return ProjectStatus.Bound;

363  void ConsumeHistoryQueue(bool afterEnqueue = false)
364  {
365  // Start consuming the queue if the first entry was just enqueued.
366  if (afterEnqueue && m_HistoryRequests.Count != 1) return;
367 
368  // Can't consume an empty queue.
369  if (m_HistoryRequests.Count == 0) return;
370 
371  var (offset, size, callback) = m_HistoryRequests.Peek();
372  // Execute next request. Discard if exception.
373  try
374  {
375  m_RevisionsService.GetRevisions(offset, size);
376  }
377  catch (Exception e)
378  {
379  Debug.LogException(e);
380  // Remove request and send failure callback.
381  m_HistoryRequests.Dequeue();
382  callback(null, null);
383  }
384  }

494  {

740  protected async Task RequestTurnOnServiceInternal()
741  {
742  Assert.IsTrue(Threading.IsMainThread, "This must be run on the main thread.");
743 
744  // Fire up the update Genesis service flag request.
745  var http = new HttpClientHandler { CookieContainer = new CookieContainer() };
746  var client = new HttpClient(http);
747 
748  var projectGuid = UnityConnect.instance.projectInfo.projectGUID;
749  var accessToken = UnityConnect.instance.GetAccessToken();
750  client.DefaultRequestHeaders.Authorization = new AuthenticationHeaderValue("Bearer", accessToken);
751  client.DefaultRequestHeaders.Accept.Add(new MediaTypeWithQualityHeaderValue("application/json"));
752  client.DefaultRequestHeaders.TryAddWithoutValidation("X-UNITY-VERSION", InternalEditorUtility.GetFullUnityVersion());
753 
754  var fullUrl = $"{UnityConnect.instance.GetConfigurationURL(CloudConfigUrl.CloudCore)}/api/projects/{projectGuid}/service_flags";
755  const string json = @"{ ""service_flags"": { ""collab"" : true} }";
756  var content = new StringContent(json, Encoding.UTF8, "application/json");
757 
758  var response = await PutAsync(client, fullUrl, content);
759 
760  // Success.
761  if (response?.StatusCode == HttpStatusCode.OK)
762  {
763  SaveAssets();
764  TurnOnCollabInternal();
765  }
766  // Error.
767  else if (response?.StatusCode == HttpStatusCode.Forbidden)
768  {
769  ShowCredentialsError();
770  }
771  else
772  {
773  ShowGeneralError();
774  }
775  }

816  IHistoryEntry RevisionToHistoryEntry(Revision revision)
817  {
818  var time = DateTimeOffset.FromUnixTimeSeconds((long)revision.timeStamp);
819  var entries = revision.entries.Select(ChangeActionToChangeEntry).ToList();
820  var status = HistoryEntryStatus.Ahead;
821  if (revision.isObtained)
822  status = HistoryEntryStatus.Behind;
823  if (revision.revisionID == m_RevisionsService.tipRevision)
824  status = HistoryEntryStatus.Current;
825  return new HistoryEntry(revision.revisionID, status, revision.author, revision.comment, time, entries);
826  }

833  static IChangeEntry ChangeActionToChangeEntry(ChangeAction action)
834  {
835  var unmerged = false;
836  var status = ChangeEntryStatus.None;
837  switch (action.action.ToLower())
838  {
839  case "added":
840  status = ChangeEntryStatus.Added;
841  break;
842  case "conflict":
843  status = ChangeEntryStatus.Unmerged;
844  unmerged = true;
845  break;
846  case "deleted":
847  status = ChangeEntryStatus.Deleted;
848  break;
849  case "ignored":
850  status = ChangeEntryStatus.Ignored;
851  break;
852  case "renamed":
853  case "moved":
854  status = ChangeEntryStatus.Renamed;
855  break;
856  case "updated":
857  status = ChangeEntryStatus.Modified;
858  break;
859  default:
860  Debug.LogError($"Unknown file status: {action.action}");
861  break;
862  }
863 
864  return new ChangeEntry(action.path, status: status, unmerged: unmerged);
865  }

874  static ChangeEntryStatus ChangeEntryStatusFromCollabState(CollabStates state)
875  {
876  if (IsCollabStateFlagSet(state, CollabStates.kCollabIgnored))
877  {
878  return ChangeEntryStatus.Ignored;
879  }
880  if (IsCollabStateFlagSet(state, CollabStates.kCollabConflicted | CollabStates.kCollabPendingMerge))
881  {
882  return ChangeEntryStatus.Unmerged;
883  }
884  if (IsCollabStateFlagSet(state, CollabStates.kCollabAddedLocal))
885  {
886  return ChangeEntryStatus.Added;
887  }
888  if (IsCollabStateFlagSet(state, CollabStates.kCollabMovedLocal))
889  {
890  return ChangeEntryStatus.Renamed;
891  }
892  if (IsCollabStateFlagSet(state, CollabStates.kCollabDeletedLocal))
893  {
894  return ChangeEntryStatus.Deleted;
895  }
