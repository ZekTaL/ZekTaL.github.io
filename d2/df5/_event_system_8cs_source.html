199  private static int RaycastComparer(RaycastResult lhs, RaycastResult rhs)
200  {
201  if (lhs.module != rhs.module)
202  {
203  var lhsEventCamera = lhs.module.eventCamera;
204  var rhsEventCamera = rhs.module.eventCamera;
205  if (lhsEventCamera != null && rhsEventCamera != null && lhsEventCamera.depth != rhsEventCamera.depth)
206  {
207  // need to reverse the standard compareTo
208  if (lhsEventCamera.depth < rhsEventCamera.depth)
209  return 1;
210  if (lhsEventCamera.depth == rhsEventCamera.depth)
211  return 0;
212 
213  return -1;
214  }
215 
216  if (lhs.module.sortOrderPriority != rhs.module.sortOrderPriority)
217  return rhs.module.sortOrderPriority.CompareTo(lhs.module.sortOrderPriority);
218 
219  if (lhs.module.renderOrderPriority != rhs.module.renderOrderPriority)
220  return rhs.module.renderOrderPriority.CompareTo(lhs.module.renderOrderPriority);
221  }
222 
223  if (lhs.sortingLayer != rhs.sortingLayer)
224  {
225  // Uses the layer value to properly compare the relative order of the layers.
226  var rid = SortingLayer.GetLayerValueFromID(rhs.sortingLayer);
227  var lid = SortingLayer.GetLayerValueFromID(lhs.sortingLayer);
228  return rid.CompareTo(lid);
229  }
230 
231  if (lhs.sortingOrder != rhs.sortingOrder)
232  return rhs.sortingOrder.CompareTo(lhs.sortingOrder);
233 
234  // comparing depth only makes sense if the two raycast results have the same root canvas (case 912396)
235  if (lhs.depth != rhs.depth && lhs.module.rootRaycaster == rhs.module.rootRaycaster)
236  return rhs.depth.CompareTo(lhs.depth);
237 
238  if (lhs.distance != rhs.distance)
239  return lhs.distance.CompareTo(rhs.distance);
240 
241  return lhs.index.CompareTo(rhs.index);
242  }
