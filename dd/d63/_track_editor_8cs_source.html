197  public string GetErrorText(TrackAsset track, UnityEngine.Object boundObject, TrackBindingErrors detectErrors)
198  {
199  if (track == null || boundObject == null)
200  return string.Empty;
201 
202  var bindingType = GetBindingType(track);
203  if (bindingType != null)
204  {
205  // bound to a prefab asset
206  if (HasFlag(detectErrors, TrackBindingErrors.PrefabBound) && PrefabUtility.IsPartOfPrefabAsset(boundObject))
207  {
208  return k_PrefabBound;
209  }
210 
211  // If we are a component, allow for bound game objects (legacy)
212  if (typeof(Component).IsAssignableFrom(bindingType))
213  {
214  var gameObject = boundObject as GameObject;
215  var component = boundObject as Component;
216  if (component != null)
217  gameObject = component.gameObject;
218 
219  // game object is bound with no component
220  if (HasFlag(detectErrors, TrackBindingErrors.NoValidComponent) && gameObject != null && component == null)
221  {
222  component = gameObject.GetComponent(bindingType);
223  if (component == null)
224  {
225  return k_NoValidComponent;
226  }
227  }
228 
229  // attached gameObject is disables (ignores Activation Track)
230  if (HasFlag(detectErrors, TrackBindingErrors.BoundGameObjectDisabled) && gameObject != null && !gameObject.activeInHierarchy)
231  {
232  return k_BoundGameObjectDisabled;
233  }
234 
235  // component is disabled
236  var behaviour = component as Behaviour;
237  if (HasFlag(detectErrors, TrackBindingErrors.BehaviourIsDisabled) && behaviour != null && !behaviour.enabled)
238  {
239  return k_RequiredComponentIsDisabled;
240  }
241 
242  // mismatched binding
243  if (HasFlag(detectErrors, TrackBindingErrors.InvalidBinding) && component != null && !bindingType.IsAssignableFrom(component.GetType()))
244  {
245  return k_InvalidBinding;
246  }
247  }
248  // Mismatched binding (non-component)
249  else if (HasFlag(detectErrors, TrackBindingErrors.InvalidBinding) && !bindingType.IsAssignableFrom(boundObject.GetType()))
250  {
251  return k_InvalidBinding;
252  }
253  }
254 
255  return string.Empty;
256  }
