644  protected override void OnPopulateMesh(VertexHelper toFill)
645  {
646  if (font == null)
647  return;
648 
649  // We don't care if we the font Texture changes while we are doing our Update.
650  // The end result of cachedTextGenerator will be valid for this instance.
651  // Otherwise we can get issues like Case 619238.
652  m_DisableFontTextureRebuiltCallback = true;
653 
654  Vector2 extents = rectTransform.rect.size;
655 
656  var settings = GetGenerationSettings(extents);
657  cachedTextGenerator.PopulateWithErrors(text, settings, gameObject);
658 
659  // Apply the offset to the vertices
660  IList<UIVertex> verts = cachedTextGenerator.verts;
661  float unitsPerPixel = 1 / pixelsPerUnit;
662  int vertCount = verts.Count;
663 
664  // We have no verts to process just return (case 1037923)
665  if (vertCount <= 0)
666  {
667  toFill.Clear();
668  return;
669  }
670 
671  Vector2 roundingOffset = new Vector2(verts[0].position.x, verts[0].position.y) * unitsPerPixel;
672  roundingOffset = PixelAdjustPoint(roundingOffset) - roundingOffset;
673  toFill.Clear();
674  if (roundingOffset != Vector2.zero)
675  {
676  for (int i = 0; i < vertCount; ++i)
677  {
678  int tempVertsIndex = i & 3;
679  m_TempVerts[tempVertsIndex] = verts[i];
680  m_TempVerts[tempVertsIndex].position *= unitsPerPixel;
681  m_TempVerts[tempVertsIndex].position.x += roundingOffset.x;
682  m_TempVerts[tempVertsIndex].position.y += roundingOffset.y;
683  if (tempVertsIndex == 3)
684  toFill.AddUIVertexQuad(m_TempVerts);
685  }
686  }
687  else
688  {
689  for (int i = 0; i < vertCount; ++i)
690  {
691  int tempVertsIndex = i & 3;
692  m_TempVerts[tempVertsIndex] = verts[i];
693  m_TempVerts[tempVertsIndex].position *= unitsPerPixel;
694  if (tempVertsIndex == 3)
695  toFill.AddUIVertexQuad(m_TempVerts);
696  }
697  }
