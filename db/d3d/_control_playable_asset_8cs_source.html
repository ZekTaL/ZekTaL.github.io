286  internal void UpdateDurationAndLoopFlag(IList<PlayableDirector> directors, IList<ParticleSystem> particleSystems)
287  {
288  if (directors.Count == 0 && particleSystems.Count == 0)
289  return;
290 
291  const double invalidDuration = double.NegativeInfinity;
292 
293  var maxDuration = invalidDuration;
294  var supportsLoop = false;
295 
296  foreach (var director in directors)
297  {
298  if (director.playableAsset != null)
299  {
300  var assetDuration = director.playableAsset.duration;
301 
302  if (director.playableAsset is TimelineAsset && assetDuration > 0.0)
303  // Timeline assets report being one tick shorter than they actually are, unless they are empty
304  assetDuration = (double)((DiscreteTime)assetDuration).OneTickAfter();
305 
306  maxDuration = Math.Max(maxDuration, assetDuration);
307  supportsLoop = supportsLoop || director.extrapolationMode == DirectorWrapMode.Loop;
308  }
309  }
310 
311  foreach (var particleSystem in particleSystems)
312  {
313  maxDuration = Math.Max(maxDuration, particleSystem.main.duration);
314  supportsLoop = supportsLoop || particleSystem.main.loop;
315  }
316 
317  m_Duration = double.IsNegativeInfinity(maxDuration) ? PlayableBinding.DefaultDuration : maxDuration;
318  m_SupportLoop = supportsLoop;
319  }
