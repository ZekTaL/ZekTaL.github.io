91  {
92  // handles child properties (Vector3 is 3 recordable properties)
93  if (property.hasChildren)
94  {
95  var iter = property.Copy();
96  var end = property.GetEndProperty(false);
97 
98  // recurse over all children properties
99  while (iter.Next(true) && !SerializedProperty.EqualContents(iter, end))
100  {
101  GatherModifications(iter, modifications);
102  }
103  }
104 
105  var isObject = property.propertyType == SerializedPropertyType.ObjectReference;
106  var isFloat = property.propertyType == SerializedPropertyType.Float ||
107  property.propertyType == SerializedPropertyType.Boolean ||
108  property.propertyType == SerializedPropertyType.Integer;
109 
110  if (isObject || isFloat)
111  {
112  var serializedObject = property.serializedObject;
113  var modification = new PropertyModification();
114 
115  modification.target = serializedObject.targetObject;
116  modification.propertyPath = property.propertyPath;
117  if (isObject)
118  {
119  modification.value = string.Empty;
120  modification.objectReference = property.objectReferenceValue;
121  }
122  else
123  {
124  modification.value = TimelineUtility.PropertyToString(property);
125  }

127  // Path for monobehaviour based - better to grab the component to get the curvebinding to allow validation
128  if (serializedObject.targetObject is Component)

210  // Helper that finds the animation clip we are recording and the relative time to that clip
211  static bool GetClipAndRelativeTime(UnityEngine.Object target, WindowState state,
212  out AnimationClip outClip, out double keyTime, out bool keyInRange)
213  {
214  const float floatToDoubleError = 0.00001f;
215  outClip = null;
216  keyTime = 0;
217  keyInRange = false;
218 
219  double startTime = 0;
220  double timeScale = 1;
221  AnimationClip clip = null;
222 
223  IPlayableAsset playableAsset = target as IPlayableAsset;
224  Component component = target as Component;
225 
226  // Handle recordable playable assets
227  if (playableAsset != null)
228  {
229  var curvesOwner = AnimatedParameterUtility.ToCurvesOwner(playableAsset, state.editSequence.asset);
230  if (curvesOwner != null)
231  {
232  if (curvesOwner.curves == null)
233  curvesOwner.CreateCurves(curvesOwner.GetUniqueRecordedClipName());
234 
235  clip = curvesOwner.curves;
236 
237  var timelineClip = curvesOwner as TimelineClip;
238  if (timelineClip != null)
239  {
240  startTime = timelineClip.start;
241  timeScale = timelineClip.timeScale;
242  }
243  }
244  }
245  // Handle recording components, including infinite clip
246  else if (component != null)
247  {
248  var asset = GetTrackForGameObject(component.gameObject, state);
249  if (asset != null)
250  {
251  clip = GetRecordingClip(asset, state, out startTime, out timeScale);
252  }
253  }
254 
255  if (clip == null)
256  return false;
257 
258  keyTime = (state.editSequence.time - startTime) * timeScale;
259  outClip = clip;
260  keyInRange = keyTime >= 0 && keyTime <= (clip.length * timeScale + floatToDoubleError);
261 
262  return true;
263  }

356  static HashSet<double> GetKeyTimes(UnityEngine.Object target, IEnumerable<PropertyModification> modifications, WindowState state)
357  {
358  var keyTimes = new HashSet<double>();
359 
360  AnimationClip animationClip;
361  double keyTime;
362  bool inRange;
363  GetClipAndRelativeTime(target, state, out animationClip, out keyTime, out inRange);
364  if (animationClip == null)
365  return keyTimes;
366 
367  var component = target as Component;
368  var playableAsset = target as IPlayableAsset;
369  var info = AnimationClipCurveCache.Instance.GetCurveInfo(animationClip);
370 
371  TimelineClip clip = null;
372  if (component != null)
373  {
374  GetTrackForGameObject(component.gameObject, state).FindRecordingClipAtTime(state.editSequence.time, out clip);
375  }
376  else if (playableAsset != null)
377  {
378  clip = FindClipWithAsset(state.editSequence.asset, playableAsset);
379  }
380 
381  foreach (var mod in modifications)
382  {
383  EditorCurveBinding temp;
384  if (HasBinding(target, mod, animationClip, out temp))
385  {
386  IEnumerable<double> keys = new HashSet<double>();
387  if (temp.isPPtrCurve)
388  {
389  var curve = info.GetObjectCurveForBinding(temp);
390  if (curve != null)
391  {
392  keys = curve.Select(x => (double)x.time);
393  }
394  }
395  else
396  {
397  var curve = info.GetCurveForBinding(temp);
398  if (curve != null)
399  {
400  keys = curve.keys.Select(x => (double)x.time);
401  }
402  }
403 
404  // Transform the times in to 'global' space using the clip
405  if (clip != null)
406  {
407  foreach (var k in keys)
408  {
409  var time = clip.FromLocalTimeUnbound(k);
410  const double eps = 1e-5;
411  if (time >= clip.start - eps && time <= clip.end + eps)
412  {
413  keyTimes.Add(time);
414  }
415  }
416  }
417  // infinite clip mode, global == local space
418  else
419  {
420  keyTimes.UnionWith(keys);
421  }
422  }
423  }
424 
425  return keyTimes;
426  }
