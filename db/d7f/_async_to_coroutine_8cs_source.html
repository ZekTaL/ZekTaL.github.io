16  // Send will process the call synchronously. If the call is processed on the main thread, we'll invoke it
17  // directly here. If the call is processed on another thread it will be queued up like POST to be executed
18  // on the main thread and it will wait. Once the main thread processes the work we can continue
19  public override void Send(SendOrPostCallback callback, object state)
20  {
21  if (m_MainThreadId == Thread.CurrentThread.ManagedThreadId)
22  {
23  callback(state);
24  }
25  else
26  {
27  using (var waitHandle = new ManualResetEvent(false))
28  {
29  lock (m_AsyncWorkQueue)
30  {
31  m_AsyncWorkQueue.Enqueue(new WorkRequest(callback, state, waitHandle));
32  }
33  waitHandle.WaitOne();
34  }
35  }
36  }
37 
38  // Post will add the call to a task list to be executed later on the main thread then work will continue asynchronously
39  public override void Post(SendOrPostCallback callback, object state)
40  {
41  lock (m_AsyncWorkQueue)
42  {
43  m_AsyncWorkQueue.Enqueue(new WorkRequest(callback, state));
44  }
45  }
46 
47  // Exec will execute tasks off the task list
48  public void Exec()
49  {
50  lock (m_AsyncWorkQueue)
51  {
52  var workCount = m_AsyncWorkQueue.Count;
53  for (var i = 0; i < workCount; i++)
54  {
55  var work = m_AsyncWorkQueue.Dequeue();
56  work.Invoke();
57  }
58  }
