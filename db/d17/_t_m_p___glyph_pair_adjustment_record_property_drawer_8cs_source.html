318  {
319  // Get a reference to the font asset

321 
322  if (fontAsset == null)
323  return;
324 
325  Glyph glyph;
326 
327  // Check if glyph is present in the atlas texture.
328  if (!fontAsset.glyphLookupTable.TryGetValue(glyphIndex, out glyph))
329  return;
330 
331  // Get the atlas index of the glyph and lookup its atlas texture
332  int atlasIndex = glyph.atlasIndex;
333  Texture2D atlasTexture = fontAsset.atlasTextures.Length > atlasIndex ? fontAsset.atlasTextures[atlasIndex] : null;
334 
335  if (atlasTexture == null)
336  return;
337 
338  Material mat;
339  if (((GlyphRasterModes)fontAsset.atlasRenderMode & GlyphRasterModes.RASTER_MODE_BITMAP) == GlyphRasterModes.RASTER_MODE_BITMAP)
340  {
341  mat = TMP_FontAssetEditor.internalBitmapMaterial;
342 
343  if (mat == null)
344  return;
345 
346  mat.mainTexture = atlasTexture;
347  }
348  else
349  {
350  mat = TMP_FontAssetEditor.internalSDFMaterial;
351 
352  if (mat == null)
353  return;
354 
355  mat.mainTexture = atlasTexture;
356  mat.SetFloat(ShaderUtilities.ID_GradientScale, fontAsset.atlasPadding + 1);
357  }
358 
359  // Draw glyph from atlas texture.
360  Rect glyphDrawPosition = new Rect(position.x, position.y + 2, 64, 60);
361 
362  GlyphRect glyphRect = glyph.glyphRect;
363 
364  int padding = fontAsset.atlasPadding;
365 
366  int glyphOriginX = glyphRect.x - padding;
367  int glyphOriginY = glyphRect.y - padding;

369  int glyphHeight = glyphRect.height + padding * 2;
370

372  float scale = glyphDrawPosition.width / normalizedHeight;
373 
374  // Compute the normalized texture coordinates

380  glyphDrawPosition.y += (glyphDrawPosition.height - glyphHeight * scale) / 2;

382  glyphDrawPosition.height = glyphHeight * scale;
