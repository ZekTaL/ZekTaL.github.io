19  // Safe for packages as we use packageInfo.resolvedPath, so this should work in library package cache as well
20  public static string[] FindPackageAssetFullPath(string assetfilter, string filefilter)
21  {
22  return AssetDatabase.FindAssets(assetfilter)
23  .Select(AssetDatabase.GUIDToAssetPath)
24  .Where(assetPath => assetPath.Contains(filefilter))
25  .Select(asset =>
26  {
27  var packageInfo = UnityEditor.PackageManager.PackageInfo.FindForAssetPath(asset);
28  return Normalize(packageInfo.resolvedPath + asset.Substring(packageInfo.assetPath.Length));
29  })
30  .ToArray();
31  }

39  public static string Normalize(string path)
40  {
41  if (string.IsNullOrEmpty(path))
42  return path;
43 
44  if (Path.DirectorySeparatorChar == WinSeparator)
45  path = path.Replace(UnixSeparator, WinSeparator);
46  if (Path.DirectorySeparatorChar == UnixSeparator)
47  path = path.Replace(WinSeparator, UnixSeparator);
48 
49  return path.Replace(string.Concat(WinSeparator, WinSeparator), WinSeparator.ToString());
50  }
51 
52  public static string NormalizeWindowsToUnix(string path)
53  {
54  if (string.IsNullOrEmpty(path))
55  return path;
56 
57  return path.Replace(WinSeparator, UnixSeparator);
58  }

60  internal static bool IsFileInProjectRootDirectory(string fileName)
61  {
62  var relative = MakeRelativeToProjectPath(fileName);
63  if (string.IsNullOrEmpty(relative))
64  return false;
65 
66  return relative == Path.GetFileName(relative);

69  // returns null if outside of the project scope
70  internal static string MakeRelativeToProjectPath(string fileName)
71  {
72  var basePath = Path.GetFullPath(Path.Combine(Application.dataPath, ".."));
73  fileName = Normalize(fileName);
74 
75  if (!Path.IsPathRooted(fileName))
76  fileName = Path.Combine(basePath, fileName);
77 
78  if (!fileName.StartsWith(basePath, StringComparison.OrdinalIgnoreCase))
79  return null;
80 
81  return fileName
82  .Substring(basePath.Length)
83  .Trim(Path.DirectorySeparatorChar);
84  }
