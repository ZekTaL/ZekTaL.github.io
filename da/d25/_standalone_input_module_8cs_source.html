294  private bool ProcessTouchEvents()
295  {
296  for (int i = 0; i < input.touchCount; ++i)
297  {
298  Touch touch = input.GetTouch(i);
299 
300  if (touch.type == TouchType.Indirect)
301  continue;
302 
303  bool released;
304  bool pressed;
305  var pointer = GetTouchPointerEventData(touch, out pressed, out released);
306 
307  ProcessTouchPress(pointer, pressed, released);
308 
309  if (!released)
310  {
311  ProcessMove(pointer);
312  ProcessDrag(pointer);
313  }
314  else
315  RemovePointerData(pointer);
316  }
317  return input.touchCount > 0;

333  // PointerDown notification
334  if (pressed)

351 
352  // search for the control that will receive the press
353  // if we can't find a press handler set the press
354  // handler to be what would receive a click.
355  var newPressed = ExecuteEvents.ExecuteHierarchy(currentOverGo, pointerEvent, ExecuteEvents.pointerDownHandler);
356 
357  var newClick = ExecuteEvents.GetEventHandler<IPointerClickHandler>(currentOverGo);
358 
359  // didnt find a press handler... search for a click handler
360  if (newPressed == null)
361  newPressed = newClick;

364

366 
367  if (newPressed == pointerEvent.lastPress)

369  var diffTime = time - pointerEvent.clickTime;
370  if (diffTime < 0.3f)
371  ++pointerEvent.clickCount;
372  else
373  pointerEvent.clickCount = 1;
374 
375  pointerEvent.clickTime = time;
376  }
377  else
378  {
379  pointerEvent.clickCount = 1;
380  }
381

384  pointerEvent.pointerClick = newClick;

389  pointerEvent.pointerDrag = ExecuteEvents.GetEventHandler<IDragHandler>(currentOverGo);
390 
391  if (pointerEvent.pointerDrag != null)

405 
406  // PointerClick and Drop events
407  if (pointerEvent.pointerClick == pointerClickHandler && pointerEvent.eligibleForClick)
408  {

481  protected bool SendMoveEventToSelectedObject()
482  {
483  float time = Time.unscaledTime;
484 
485  Vector2 movement = GetRawMoveVector();
486  if (Mathf.Approximately(movement.x, 0f) && Mathf.Approximately(movement.y, 0f))
487  {
488  m_ConsecutiveMoveCount = 0;
489  return false;
490  }
491 
492  bool similarDir = (Vector2.Dot(movement, m_LastMoveVector) > 0);
493 
494  // If direction didn't change at least 90 degrees, wait for delay before allowing consequtive event.
495  if (similarDir && m_ConsecutiveMoveCount == 1)
496  {
497  if (time <= m_PrevActionTime + m_RepeatDelay)
498  return false;
499  }
500  // If direction changed at least 90 degree, or we already had the delay, repeat at repeat rate.
501  else
502  {
503  if (time <= m_PrevActionTime + 1f / m_InputActionsPerSecond)
504  return false;
505  }
506 
507  var axisEventData = GetAxisEventData(movement.x, movement.y, 0.6f);
508 
509  if (axisEventData.moveDir != MoveDirection.None)
510  {
511  ExecuteEvents.Execute(eventSystem.currentSelectedGameObject, axisEventData, ExecuteEvents.moveHandler);
512  if (!similarDir)
513  m_ConsecutiveMoveCount = 0;
514  m_ConsecutiveMoveCount++;
515  m_PrevActionTime = time;
516  m_LastMoveVector = movement;
517  }
518  else
519  {
520  m_ConsecutiveMoveCount = 0;
521  }
522 
523  return axisEventData.used;
524  }
