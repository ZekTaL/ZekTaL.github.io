955  private void GenerateSprite(VertexHelper vh, bool lPreserveAspect)
956  {
957  var spriteSize = new Vector2(activeSprite.rect.width, activeSprite.rect.height);
958 
959  // Covert sprite pivot into normalized space.
960  var spritePivot = activeSprite.pivot / spriteSize;
961  var rectPivot = rectTransform.pivot;
962  Rect r = GetPixelAdjustedRect();
963 
964  if (lPreserveAspect & spriteSize.sqrMagnitude > 0.0f)
965  {
966  PreserveSpriteAspectRatio(ref r, spriteSize);
967  }
968 
969  var drawingSize = new Vector2(r.width, r.height);
970  var spriteBoundSize = activeSprite.bounds.size;
971 
972  // Calculate the drawing offset based on the difference between the two pivots.
973  var drawOffset = (rectPivot - spritePivot) * drawingSize;
974 
975  var color32 = color;
976  vh.Clear();
977 
978  Vector2[] vertices = activeSprite.vertices;
979  Vector2[] uvs = activeSprite.uv;
980  for (int i = 0; i < vertices.Length; ++i)
981  {
982  vh.AddVert(new Vector3((vertices[i].x / spriteBoundSize.x) * drawingSize.x - drawOffset.x, (vertices[i].y / spriteBoundSize.y) * drawingSize.y - drawOffset.y), color32, new Vector2(uvs[i].x, uvs[i].y));
983  }
984 
985  UInt16[] triangles = activeSprite.triangles;
986  for (int i = 0; i < triangles.Length; i += 3)
987  {
988  vh.AddTriangle(triangles[i + 0], triangles[i + 1], triangles[i + 2]);
989  }
990  }

1076  void GenerateTiledSprite(VertexHelper toFill)
1077  {
1078  Vector4 outer, inner, border;
1079  Vector2 spriteSize;
1080 
1081  if (activeSprite != null)
1082  {
1083  outer = Sprites.DataUtility.GetOuterUV(activeSprite);
1084  inner = Sprites.DataUtility.GetInnerUV(activeSprite);
1085  border = activeSprite.border;
1086  spriteSize = activeSprite.rect.size;
1087  }
1088  else
1089  {
1090  outer = Vector4.zero;
1091  inner = Vector4.zero;
1092  border = Vector4.zero;
1093  spriteSize = Vector2.one * 100;
1094  }
1095 
1096  Rect rect = GetPixelAdjustedRect();
1097  float tileWidth = (spriteSize.x - border.x - border.z) / multipliedPixelsPerUnit;
1098  float tileHeight = (spriteSize.y - border.y - border.w) / multipliedPixelsPerUnit;
1099 
1100  border = GetAdjustedBorders(border / multipliedPixelsPerUnit, rect);
1101 
1102  var uvMin = new Vector2(inner.x, inner.y);
1103  var uvMax = new Vector2(inner.z, inner.w);
1104 
1105  // Min to max max range for tiled region in coordinates relative to lower left corner.
1106  float xMin = border.x;
1107  float xMax = rect.width - border.z;
1108  float yMin = border.y;
1109  float yMax = rect.height - border.w;
1110 
1111  toFill.Clear();
1112  var clipped = uvMax;
1113 
1114  // if either width is zero we cant tile so just assume it was the full width.
1115  if (tileWidth <= 0)
1116  tileWidth = xMax - xMin;
1117 
1118  if (tileHeight <= 0)
1119  tileHeight = yMax - yMin;
1120 
1121  if (activeSprite != null && (hasBorder || activeSprite.packed || activeSprite.texture.wrapMode != TextureWrapMode.Repeat))
1122  {
1123  // Sprite has border, or is not in repeat mode, or cannot be repeated because of packing.
1124  // We cannot use texture tiling so we will generate a mesh of quads to tile the texture.
1125 
1126  // Evaluate how many vertices we will generate. Limit this number to something sane,
1127  // especially since meshes can not have more than 65000 vertices.
1128 
1129  long nTilesW = 0;
1130  long nTilesH = 0;
1131  if (m_FillCenter)
1132  {
1133  nTilesW = (long)Math.Ceiling((xMax - xMin) / tileWidth);
1134  nTilesH = (long)Math.Ceiling((yMax - yMin) / tileHeight);
1135 
1136  double nVertices = 0;
1137  if (hasBorder)
1138  {
1139  nVertices = (nTilesW + 2.0) * (nTilesH + 2.0) * 4.0; // 4 vertices per tile
1140  }
1141  else
1142  {
1143  nVertices = nTilesW * nTilesH * 4.0; // 4 vertices per tile
1144  }
1145 
1146  if (nVertices > 65000.0)
1147  {
1148  Debug.LogError("Too many sprite tiles on Image \"" + name + "\". The tile size will be increased. To remove the limit on the number of tiles, set the Wrap mode to Repeat in the Image Import Settings", this);
1149 
1150  double maxTiles = 65000.0 / 4.0; // Max number of vertices is 65000; 4 vertices per tile.
1151  double imageRatio;
1152  if (hasBorder)
1153  {
1154  imageRatio = (nTilesW + 2.0) / (nTilesH + 2.0);
1155  }
1156  else
1157  {
1158  imageRatio = (double)nTilesW / nTilesH;
1159  }
1160 
1161  double targetTilesW = Math.Sqrt(maxTiles / imageRatio);
1162  double targetTilesH = targetTilesW * imageRatio;
1163  if (hasBorder)
1164  {
1165  targetTilesW -= 2;
1166  targetTilesH -= 2;
1167  }
1168 
1169  nTilesW = (long)Math.Floor(targetTilesW);
1170  nTilesH = (long)Math.Floor(targetTilesH);
1171  tileWidth = (xMax - xMin) / nTilesW;
1172  tileHeight = (yMax - yMin) / nTilesH;
1173  }
1174  }
1175  else
1176  {
1177  if (hasBorder)
1178  {
1179  // Texture on the border is repeated only in one direction.
1180  nTilesW = (long)Math.Ceiling((xMax - xMin) / tileWidth);
1181  nTilesH = (long)Math.Ceiling((yMax - yMin) / tileHeight);
1182  double nVertices = (nTilesH + nTilesW + 2.0 /*corners*/) * 2.0 /*sides*/ * 4.0 /*vertices per tile*/;
1183  if (nVertices > 65000.0)
1184  {
1185  Debug.LogError("Too many sprite tiles on Image \"" + name + "\". The tile size will be increased. To remove the limit on the number of tiles, set the Wrap mode to Repeat in the Image Import Settings", this);
1186 
1187  double maxTiles = 65000.0 / 4.0; // Max number of vertices is 65000; 4 vertices per tile.
1188  double imageRatio = (double)nTilesW / nTilesH;
1189  double targetTilesW = (maxTiles - 4 /*corners*/) / (2 * (1.0 + imageRatio));
1190  double targetTilesH = targetTilesW * imageRatio;

1344  private Vector4 GetAdjustedBorders(Vector4 border, Rect adjustedRect)
1345  {
1346  Rect originalRect = rectTransform.rect;
1347 
1348  for (int axis = 0; axis <= 1; axis++)
1349  {
1350  float borderScaleRatio;
1351 
1352  // The adjusted rect (adjusted for pixel correctness)
1353  // may be slightly larger than the original rect.
1354  // Adjust the border to match the adjustedRect to avoid
1355  // small gaps between borders (case 833201).
1356  if (originalRect.size[axis] != 0)
1357  {
1358  borderScaleRatio = adjustedRect.size[axis] / originalRect.size[axis];
1359  border[axis] *= borderScaleRatio;
1360  border[axis + 2] *= borderScaleRatio;
1361  }
1362

1364  // In order to avoid artefacts with overlapping borders, we scale the borders down to fit.
1365  float combinedBorders = border[axis] + border[axis + 2];
1366  if (adjustedRect.size[axis] < combinedBorders && combinedBorders != 0)
1367  {
1368  borderScaleRatio = adjustedRect.size[axis] / combinedBorders;
1369  border[axis] *= borderScaleRatio;
1370  border[axis + 2] *= borderScaleRatio;
1371  }
1372  }
1373  return border;

1587  {
