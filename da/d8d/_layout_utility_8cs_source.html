142  public static float GetLayoutProperty(RectTransform rect, System.Func<ILayoutElement, float> property, float defaultValue, out ILayoutElement source)
143  {
144  source = null;
145  if (rect == null)
146  return 0;
147  float min = defaultValue;
148  int maxPriority = System.Int32.MinValue;
149  var components = ListPool<Component>.Get();
150  rect.GetComponents(typeof(ILayoutElement), components);
151 
152  var componentsCount = components.Count;
153  for (int i = 0; i < componentsCount; i++)
154  {
155  var layoutComp = components[i] as ILayoutElement;
156  if (layoutComp is Behaviour && !((Behaviour)layoutComp).isActiveAndEnabled)
157  continue;
158 
159  int priority = layoutComp.layoutPriority;
160  // If this layout components has lower priority than a previously used, ignore it.
161  if (priority < maxPriority)
162  continue;
163  float prop = property(layoutComp);
164  // If this layout property is set to a negative value, it means it should be ignored.
165  if (prop < 0)
166  continue;
167 
168  // If this layout component has higher priority than all previous ones,
169  // overwrite with this one's value.
170  if (priority > maxPriority)
171  {
172  min = prop;
173  maxPriority = priority;
174  source = layoutComp;
175  }
176  // If the layout component has the same priority as a previously used,
177  // use the largest of the values with the same priority.
178  else if (prop > min)
179  {
180  min = prop;
181  source = layoutComp;
182  }
183  }
184 
185  ListPool<Component>.Release(components);
186  return min;
187  }
