169  void AddSerializedPlayableModifications(IPlayableAsset asset, AnimationClip clip)
170  {
171  var obj = asset as Object;
172  if (obj == null)
173  return;
174 
175  var driver = WindowState.previewDriver;
176  if (driver == null || !AnimationMode.InAnimationMode(driver))
177  return;
178 
179  var serializedObj = new SerializedObject(obj);
180  var bindings = AnimationClipCurveCache.Instance.GetCurveInfo(clip).bindings;
181  var fields = AnimatedParameterUtility.GetScriptPlayableFields(asset);
182 
183  // go through each binding and offset using the field name
184  // so the modification system can find the particle object using the asset as a root
185  foreach (var b in bindings)
186  {
187  foreach (var f in fields)
188  {
189  var propertyPath = f.Name + "." + b.propertyName;
190  if (serializedObj.FindProperty(propertyPath) != null)
191  {
192  DrivenPropertyManager.RegisterProperty(driver, obj, propertyPath);
193  break;
194  }
195  }
196  }
