127  {
128  if (canvas == null)
129  return;
130 
131  var canvasGraphics = GraphicRegistry.GetRaycastableGraphicsForCanvas(canvas);
132  if (canvasGraphics == null || canvasGraphics.Count == 0)
133  return;
134 
135  int displayIndex;
136  var currentEventCamera = eventCamera; // Property can call Camera.main, so cache the reference
137 
138  if (canvas.renderMode == RenderMode.ScreenSpaceOverlay || currentEventCamera == null)
139  displayIndex = canvas.targetDisplay;
140  else
141  displayIndex = currentEventCamera.targetDisplay;
142 
143  var eventPosition = Display.RelativeMouseAt(eventData.position);
144  if (eventPosition != Vector3.zero)
145  {
146  // We support multiple display and display identification based on event position.
147 
148  int eventDisplayIndex = (int)eventPosition.z;
149 
150  // Discard events that are not part of this display so the user does not interact with multiple displays at once.
151  if (eventDisplayIndex != displayIndex)
152  return;
153  }
154  else
155  {
156  // The multiple display system is not supported on all platforms, when it is not supported the returned position
157  // will be all zeros so when the returned index is 0 we will default to the event data to be safe.
158  eventPosition = eventData.position;
159 
160  // We dont really know in which display the event occured. We will process the event assuming it occured in our display.
161  }
162 
163  // Convert to view space
164  Vector2 pos;
165  if (currentEventCamera == null)
166  {
167  // Multiple display support only when not the main display. For display 0 the reported
168  // resolution is always the desktops resolution since its part of the display API,
169  // so we use the standard none multiple display method. (case 741751)
170  float w = Screen.width;
171  float h = Screen.height;
172  if (displayIndex > 0 && displayIndex < Display.displays.Length)
173  {
174  w = Display.displays[displayIndex].systemWidth;
175  h = Display.displays[displayIndex].systemHeight;
176  }
177  pos = new Vector2(eventPosition.x / w, eventPosition.y / h);
178  }
179  else
180  pos = currentEventCamera.ScreenToViewportPoint(eventPosition);
181 
182  // If it's outside the camera's viewport, do nothing
183  if (pos.x < 0f || pos.x > 1f || pos.y < 0f || pos.y > 1f)
184  return;
185 
186  float hitDistance = float.MaxValue;
187 
188  Ray ray = new Ray();
189 
190  if (currentEventCamera != null)
191  ray = currentEventCamera.ScreenPointToRay(eventPosition);
192 
193  if (canvas.renderMode != RenderMode.ScreenSpaceOverlay && blockingObjects != BlockingObjects.None)
194  {
195  float distanceToClipPlane = 100.0f;
196 
197  if (currentEventCamera != null)
198  {
199  float projectionDirection = ray.direction.z;
200  distanceToClipPlane = Mathf.Approximately(0.0f, projectionDirection)
201  ? Mathf.Infinity
202  : Mathf.Abs((currentEventCamera.farClipPlane - currentEventCamera.nearClipPlane) / projectionDirection);
203  }
204 #if PACKAGE_PHYSICS
205  if (blockingObjects == BlockingObjects.ThreeD || blockingObjects == BlockingObjects.All)
206  {
207  if (ReflectionMethodsCache.Singleton.raycast3D != null)
208  {
209  var hits = ReflectionMethodsCache.Singleton.raycast3DAll(ray, distanceToClipPlane, (int)m_BlockingMask);
210  if (hits.Length > 0)
211  hitDistance = hits[0].distance;
212  }
213  }
214 #endif
215 #if PACKAGE_PHYSICS2D
216  if (blockingObjects == BlockingObjects.TwoD || blockingObjects == BlockingObjects.All)
217  {
218  if (ReflectionMethodsCache.Singleton.raycast2D != null)
219  {
220  var hits = ReflectionMethodsCache.Singleton.getRayIntersectionAll(ray, distanceToClipPlane, (int)m_BlockingMask);
221  if (hits.Length > 0)
222  hitDistance = hits[0].distance;
223  }
224  }
225 #endif
226  }

229 
230  Raycast(canvas, currentEventCamera, eventPosition, canvasGraphics, m_RaycastResults);
231 
232  int totalCount = m_RaycastResults.Count;
233  for (var index = 0; index < totalCount; index++)
234  {
235  var go = m_RaycastResults[index].gameObject;
236  bool appendGraphic = true;
237 
238  if (ignoreReversedGraphics)
239  {
240  if (currentEventCamera == null)
241  {
242  // If we dont have a camera we know that we should always be facing forward
243  var dir = go.transform.rotation * Vector3.forward;
244  appendGraphic = Vector3.Dot(Vector3.forward, dir) > 0;
245  }
246  else
247  {
248  // If we have a camera compare the direction against the cameras forward.
249  var cameraForward = currentEventCamera.transform.rotation * Vector3.forward * currentEventCamera.nearClipPlane;
250  appendGraphic = Vector3.Dot(go.transform.position - currentEventCamera.transform.position - cameraForward, go.transform.forward) >= 0;
251  }
252  }
253 
254  if (appendGraphic)
255  {
256  float distance = 0;
257  Transform trans = go.transform;
258  Vector3 transForward = trans.forward;
259 
260  if (currentEventCamera == null || canvas.renderMode == RenderMode.ScreenSpaceOverlay)
261  distance = 0;
262  else
263  {
264  // http://geomalgorithms.com/a06-_intersect-2.html
265  distance = (Vector3.Dot(transForward, trans.position - ray.origin) / Vector3.Dot(transForward, ray.direction));
266 
267  // Check to see if the go is behind the camera.
268  if (distance < 0)
269  continue;
270  }
271 
272  if (distance >= hitDistance)
273  continue;
274 
275  var castResult = new RaycastResult
276  {
277  gameObject = go,
278  module = this,
279  distance = distance,
280  screenPosition = eventPosition,
281  displayIndex = displayIndex,
282  index = resultAppendList.Count,
283  depth = m_RaycastResults[index].depth,
284  sortingLayer = canvas.sortingLayerID,
285  sortingOrder = canvas.sortingOrder,
286  worldPosition = ray.origin + ray.direction * distance,
287  worldNormal = -transForward
288  };
289  resultAppendList.Add(castResult);
290  }
291  }

320  private static void Raycast(Canvas canvas, Camera eventCamera, Vector2 pointerPosition, IList<Graphic> foundGraphics, List<Graphic> results)
321  {
322  // Necessary for the event system
323  int totalCount = foundGraphics.Count;
324  for (int i = 0; i < totalCount; ++i)
325  {
326  Graphic graphic = foundGraphics[i];
327 
328  // -1 means it hasn't been processed by the canvas, which means it isn't actually drawn
329  if (!graphic.raycastTarget || graphic.canvasRenderer.cull || graphic.depth == -1)
330  continue;
331 
332  if (!RectTransformUtility.RectangleContainsScreenPoint(graphic.rectTransform, pointerPosition, eventCamera, graphic.raycastPadding))
333  continue;
334 
335  if (eventCamera != null && eventCamera.WorldToScreenPoint(graphic.rectTransform.position).z > eventCamera.farClipPlane)
336  continue;
337 
338  if (graphic.Raycast(pointerPosition, eventCamera))
339  {
340  s_SortedGraphics.Add(graphic);
341  }
342  }
343 
344  s_SortedGraphics.Sort((g1, g2) => g2.depth.CompareTo(g1.depth));
345  totalCount = s_SortedGraphics.Count;
346  for (int i = 0; i < totalCount; ++i)
347  results.Add(s_SortedGraphics[i]);
348 
349  s_SortedGraphics.Clear();
350  }
351  }
