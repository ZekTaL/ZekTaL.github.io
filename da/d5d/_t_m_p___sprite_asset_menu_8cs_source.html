53  internal static void UpdateSpriteAsset(TMP_SpriteAsset spriteAsset)
54  {
55  // Get a list of all the sprites contained in the texture referenced by the sprite asset.
56  // This only works if the texture is set to sprite mode.
57  string filePath = AssetDatabase.GetAssetPath(spriteAsset.spriteSheet);
58 
59  if (string.IsNullOrEmpty(filePath))
60  return;
61 
62  // Get all the sprites defined in the sprite sheet texture referenced by this sprite asset.
63  Sprite[] sprites = AssetDatabase.LoadAllAssetsAtPath(filePath).Select(x => x as Sprite).Where(x => x != null).ToArray();
64 
65  // Return if sprite sheet texture does not have any sprites defined in it.
66  if (sprites.Length == 0)
67  {
68  Debug.Log("Sprite Asset <color=#FFFF80>[" + spriteAsset.name + "]</color>'s atlas texture does not appear to have any sprites defined in it. Use the Unity Sprite Editor to define sprites for this texture.", spriteAsset.spriteSheet);
69  return;
70  }
71 
72  List<TMP_SpriteGlyph> spriteGlyphTable = spriteAsset.spriteGlyphTable;
73 
74  // Find available glpyh indexes
75  uint[] existingGlyphIndexes = spriteGlyphTable.Select(x => x.index).ToArray();
76  List<uint> availableGlyphIndexes = new List<uint>();
77 
78  uint lastGlyphIndex = existingGlyphIndexes.Length > 0 ? existingGlyphIndexes.Last() : 0;
79  int elementIndex = 0;
80  for (uint i = 0; i < lastGlyphIndex; i++)
81  {
82  uint existingGlyphIndex = existingGlyphIndexes[elementIndex];
83 
84  if (i == existingGlyphIndex)
85  elementIndex += 1;
86  else
87  availableGlyphIndexes.Add(i);
88  }
89 
90  // Iterate over sprites contained in the updated sprite sheet to identify new and / or modified sprites.
91  for (int i = 0; i < sprites.Length; i++)
92  {
93  Sprite sprite = sprites[i];
94 
95  // Check if current sprites is already contained in the sprite glyph table of the sprite asset.
96  TMP_SpriteGlyph spriteGlyph = spriteGlyphTable.FirstOrDefault(x => x.sprite == sprite);
97 
98  if (spriteGlyph != null)
99  {
100  // update existing sprite glyph
101  if (spriteGlyph.glyphRect.x != sprite.rect.x || spriteGlyph.glyphRect.y != sprite.rect.y || spriteGlyph.glyphRect.width != sprite.rect.width || spriteGlyph.glyphRect.height != sprite.rect.height)
102  spriteGlyph.glyphRect = new GlyphRect(sprite.rect);
103  }
104  else
105  {
106  TMP_SpriteCharacter spriteCharacter;
107 
108  // Check if this sprite potentially exists under the same name in the sprite character table.
109  if (spriteAsset.spriteCharacterTable != null && spriteAsset.spriteCharacterTable.Count > 0)
110  {
111  spriteCharacter = spriteAsset.spriteCharacterTable.FirstOrDefault(x => x.name == sprite.name);
112  spriteGlyph = spriteCharacter != null ? spriteGlyphTable[(int)spriteCharacter.glyphIndex] : null;
113 
114  if (spriteGlyph != null)
115  {
116  // Update sprite reference and data
117  spriteGlyph.sprite = sprite;
118 
119  if (spriteGlyph.glyphRect.x != sprite.rect.x || spriteGlyph.glyphRect.y != sprite.rect.y || spriteGlyph.glyphRect.width != sprite.rect.width || spriteGlyph.glyphRect.height != sprite.rect.height)
120  spriteGlyph.glyphRect = new GlyphRect(sprite.rect);
121  }
122  }
123 
124  // Add new Sprite Glyph to the table
125  spriteGlyph = new TMP_SpriteGlyph();
126 
127  // Get available glyph index
128  if (availableGlyphIndexes.Count > 0)
129  {
130  spriteGlyph.index = availableGlyphIndexes[0];
131  availableGlyphIndexes.RemoveAt(0);
132  }
133  else
134  spriteGlyph.index = (uint)spriteGlyphTable.Count;
135 
136  spriteGlyph.metrics = new GlyphMetrics(sprite.rect.width, sprite.rect.height, -sprite.pivot.x, sprite.rect.height - sprite.pivot.y, sprite.rect.width);
137  spriteGlyph.glyphRect = new GlyphRect(sprite.rect);
138  spriteGlyph.scale = 1.0f;
139  spriteGlyph.sprite = sprite;
140 
141  spriteGlyphTable.Add(spriteGlyph);
142 
143  spriteCharacter = new TMP_SpriteCharacter(0xFFFE, spriteGlyph);
144  spriteCharacter.name = sprite.name;
145  spriteCharacter.scale = 1.0f;
146 
147  spriteAsset.spriteCharacterTable.Add(spriteCharacter);
148  }
149  }
150 
151  // Update Sprite Character Table to replace unicode 0x0 by 0xFFFE
152  for (int i = 0; i < spriteAsset.spriteCharacterTable.Count; i++)
153  {
154  TMP_SpriteCharacter spriteCharacter = spriteAsset.spriteCharacterTable[i];
155  if (spriteCharacter.unicode == 0)
156  spriteCharacter.unicode = 0xFFFE;
157  }
158 
159  // Sort glyph table by glyph index
160  spriteAsset.SortGlyphTable();
161  spriteAsset.UpdateLookupTables();
162  TMPro_EventManager.ON_SPRITE_ASSET_PROPERTY_CHANGED(true, spriteAsset);

329  string filePath = AssetDatabase.GetAssetPath(spriteAsset.spriteSheet);
330 
331  // Get all the Sprites sorted Left to Right / Top to Bottom
332  Sprite[] sprites = AssetDatabase.LoadAllAssetsAtPath(filePath).Select(x => x as Sprite).Where(x => x != null).OrderByDescending(x => x.rect.y).ThenBy(x => x.rect.x).ToArray();
333 
334  for (int i = 0; i < sprites.Length; i++)
335  {
336  Sprite sprite = sprites[i];
337 
338  // Check if the sprite is already contained in the SpriteInfoList
339  int index = -1;
340  if (spriteAsset.spriteInfoList.Count > i && spriteAsset.spriteInfoList[i].sprite != null)
341  index = spriteAsset.spriteInfoList.FindIndex(item => item.sprite.GetInstanceID() == sprite.GetInstanceID());
342 
343  // Use existing SpriteInfo if it already exists
344  TMP_Sprite spriteInfo = index == -1 ? new TMP_Sprite() : spriteAsset.spriteInfoList[index];
345 
346  Rect spriteRect = sprite.rect;
347  spriteInfo.x = spriteRect.x;
348  spriteInfo.y = spriteRect.y;
349  spriteInfo.width = spriteRect.width;
350  spriteInfo.height = spriteRect.height;
351 
352  // Get Sprite Pivot
353  Vector2 pivot = new Vector2(0 - (sprite.bounds.min.x) / (sprite.bounds.extents.x * 2), 0 - (sprite.bounds.min.y) / (sprite.bounds.extents.y * 2));
354 
355  // The position of the pivot influences the Offset position.
356  spriteInfo.pivot = new Vector2(0 - pivot.x * spriteRect.width, spriteRect.height - pivot.y * spriteRect.height);
357 
358  if (index == -1)
359  {
360  // Find the next available index for this Sprite
361  int[] ids = spriteAsset.spriteInfoList.Select(item => item.id).ToArray();
362 
363  int id = 0;
364  for (int j = 0; j < ids.Length; j++ )
365  {
366  if (ids[0] != 0) break;
367 
368  if (j > 0 && (ids[j] - ids[j - 1]) > 1)
369  {
370  id = ids[j - 1] + 1;
371  break;
372  }
373 
374  id = j + 1;
375  }
376 
377  spriteInfo.sprite = sprite;
378  spriteInfo.name = sprite.name;
379  spriteInfo.hashCode = TMP_TextUtilities.GetSimpleHashCode(spriteInfo.name);
380  spriteInfo.id = id;
381  spriteInfo.xAdvance = spriteRect.width;
382  spriteInfo.scale = 1.0f;
383 
384  spriteInfo.xOffset = spriteInfo.pivot.x;
385  spriteInfo.yOffset = spriteInfo.pivot.y;
386 
387  spriteAsset.spriteInfoList.Add(spriteInfo);
388 
389  // Sort the Sprites by ID
390  spriteAsset.spriteInfoList = spriteAsset.spriteInfoList.OrderBy(s => s.id).ToList();
391  }
392  else
