35  public static void CloneExposedReferences(ScriptableObject clone, IExposedPropertyTable sourceTable, IExposedPropertyTable destTable)
36  {
37  var cloneObject = new SerializedObject(clone);
38  SerializedProperty prop = cloneObject.GetIterator();
39  while (prop.Next(true))
40  {
41  if (prop.propertyType == SerializedPropertyType.ExposedReference)
42  {
43  var exposedNameProp = prop.FindPropertyRelative("exposedName");
44  var sourceKey = exposedNameProp.stringValue;
45  var destKey = sourceKey;
46 
47  if (!IsExposedReferenceExplicitlyNamed(sourceKey))
48  destKey = GenerateExposedReferenceName();
49 
50  exposedNameProp.stringValue = destKey;
51 
52  var requiresCopy = sourceTable != destTable || sourceKey != destKey;
53  if (requiresCopy && sourceTable != null && destTable != null)
54  {
55  var valid = false;
56  var target = sourceTable.GetReferenceValue(sourceKey, out valid);
57  if (valid && target != null)
58  {
59  var existing = destTable.GetReferenceValue(destKey, out valid);
60  if (!valid || existing != target)
61  {
62  var destTableObj = destTable as UnityEngine.Object;
63  if (destTableObj != null)
64  TimelineUndo.PushUndo(destTableObj, "Create Clip");
65  destTable.SetReferenceValue(destKey, target);
66  }
67  }
68  }
69  }
70  }
71  cloneObject.ApplyModifiedPropertiesWithoutUndo();
72  }

261  var newClip = Object.Instantiate(editorClip).clip;

263  // perform fix ups for what Instantiate cannot properly detect
264  SelectionManager.Remove(newClip);

485  // Retrieves the starting point of each loop of a clip, relative to the start of the clip
486  // Note that if clip-in is bigger than the loopDuration, negative times will be added
487  public static double[] GetLoopTimes(TimelineClip clip)
488  {
489  if (!HasUsableAssetDuration(clip))
490  return new[] {-clip.clipIn / clip.timeScale};
491 
492  var times = new List<double>();
493  double loopDuration = GetLoopDuration(clip);
494 
495  if (loopDuration <= TimeUtility.kTimeEpsilon)
496  return new double[] {};
497 
498 
499  double start = -clip.clipIn / clip.timeScale;
500  double end = start + loopDuration;
501 
502  times.Add(start);
503  while (end < clip.duration - WindowState.kTimeEpsilon)
504  {
505  times.Add(end);
506  end += loopDuration;
507  }
508 
509  return times.ToArray();
510  }

782  public static void RangeSelect<T>(IList<T> totalCollection, IList<T> currentSelection, T clickedItem, Action<T> selector, Action<T> remover) where T : class
783  {
784  var firstSelect = currentSelection.FirstOrDefault();
785  if (firstSelect == null)
786  {
787  selector(clickedItem);
788  return;
789  }
790 
791  var idxFirstSelect = totalCollection.IndexOf(firstSelect);
792  var idxLastSelect = totalCollection.IndexOf(currentSelection.Last());

794 
795  //case 927807: selection is invalid
796  if (idxFirstSelect < 0)
797  {
798  SelectionManager.Clear();
799  selector(clickedItem);
800  return;
801  }

803  // Expand the selection between the first selected clip and clicked clip (insertion order is important)
804  if (idxFirstSelect < idxClicked)
805  for (var i = idxFirstSelect; i <= idxClicked; ++i)
806  selector(totalCollection[i]);
807  else
808  for (var i = idxFirstSelect; i >= idxClicked; --i)
809  selector(totalCollection[i]);
810 
811  // If clicked inside the selected range, shrink the selection between the the click and last selected clip
812  if (Math.Min(idxFirstSelect, idxLastSelect) < idxClicked && idxClicked < Math.Max(idxFirstSelect, idxLastSelect))
813  for (var i = Math.Min(idxLastSelect, idxClicked); i <= Math.Max(idxLastSelect, idxClicked); ++i)
814  remover(totalCollection[i]);
815 
816  // Ensure clicked clip is selected
817  selector(clickedItem);
