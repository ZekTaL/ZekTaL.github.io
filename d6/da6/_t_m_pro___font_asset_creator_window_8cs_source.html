335  {

337  string[] sequences = sequence.Split(',');
338 
339  foreach (string seq in sequences)
340  {
341  string[] s1 = seq.Split('-');
342 
343  if (s1.Length == 1)
344  try
345  {
346  unicodeList.Add(uint.Parse(s1[0]));
347  }
348  catch
349  {
350  Debug.Log("No characters selected or invalid format.");
351  }

355  {
356  unicodeList.Add(j);
