779  m_Template.GetComponent<Canvas>().sortingLayerID = rootCanvas.sortingLayerID;

783  m_Dropdown.name = "Dropdown List";
784  m_Dropdown.SetActive(true);

796  RectTransform contentRectTransform = content.transform as RectTransform;
797  itemTemplate.rectTransform.gameObject.SetActive(true);

846  Vector2 sizeDelta = contentRectTransform.sizeDelta;
847  sizeDelta.y = itemSize.y * m_Items.Count + offsetMin.y - offsetMax.y;
848  contentRectTransform.sizeDelta = sizeDelta;
