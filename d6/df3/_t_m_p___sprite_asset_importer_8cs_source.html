167  private static void PopulateSpriteTables(TexturePacker_JsonArray.SpriteDataObject spriteDataObject, List<TMP_SpriteCharacter> spriteCharacterTable, List<TMP_SpriteGlyph> spriteGlyphTable)
168  {
169  List<TexturePacker_JsonArray.Frame> importedSprites = spriteDataObject.frames;
170 
171  float atlasHeight = spriteDataObject.meta.size.h;
172 
173  for (int i = 0; i < importedSprites.Count; i++)
174  {
175  TexturePacker_JsonArray.Frame spriteData = importedSprites[i];
176 
177  TMP_SpriteGlyph spriteGlyph = new TMP_SpriteGlyph();
178  spriteGlyph.index = (uint)i;
179 
180  spriteGlyph.metrics = new GlyphMetrics((int)spriteData.frame.w, (int)spriteData.frame.h, -spriteData.frame.w * spriteData.pivot.x, spriteData.frame.h * spriteData.pivot.y, (int)spriteData.frame.w);
181  spriteGlyph.glyphRect = new GlyphRect((int)spriteData.frame.x, (int)(atlasHeight - spriteData.frame.h - spriteData.frame.y), (int)spriteData.frame.w, (int)spriteData.frame.h);
182  spriteGlyph.scale = 1.0f;
183 
184  spriteGlyphTable.Add(spriteGlyph);
185 
186  TMP_SpriteCharacter spriteCharacter = new TMP_SpriteCharacter(0, spriteGlyph);
187  spriteCharacter.name = spriteData.filename.Split('.')[0];
188  spriteCharacter.unicode = 0xFFFE;
189  spriteCharacter.scale = 1.0f;
190 
191  spriteCharacterTable.Add(spriteCharacter);
