119  // We only use this to find analyzers, we do not need to load this assembly anymore
120  private string GetExtensionPath()
121  {
122  if (VisualStudioEditor.IsWindows)
123  {
124  const string extensionName = "Visual Studio Tools for Unity";
125  const string extensionAssembly = "SyntaxTree.VisualStudio.Unity.dll";
126 
127  var vsDirectory = IOPath.GetDirectoryName(Path);
128  var vstuDirectory = IOPath.Combine(vsDirectory, "Extensions", "Microsoft", extensionName);
129 
130  if (File.Exists(IOPath.Combine(vstuDirectory, extensionAssembly)))
131  return vstuDirectory;
132  }
133 
134  if (VisualStudioEditor.IsOSX)
135  {
136  const string addinName = "MonoDevelop.Unity";
137  const string addinAssembly = addinName + ".dll";
138 
139  // user addins repository
140  var localAddins = IOPath.Combine(
141  Environment.GetFolderPath(Environment.SpecialFolder.Personal),
142  $"Library/Application Support/VisualStudio/${Version.Major}.0" + "/LocalInstall/Addins");
143 
144  // In the user addins repository, the addins are suffixed by their versions, like `MonoDevelop.Unity.1.0`
145  // When installing another local user addin, MD will remove files inside the folder
146  // So we browse all VSTUM addins, and return the one with an addin assembly
147  if (Directory.Exists(localAddins))
148  {
149  foreach (var folder in Directory.GetDirectories(localAddins, addinName + "*", SearchOption.TopDirectoryOnly))
150  {
151  if (File.Exists(IOPath.Combine(folder, addinAssembly)))
152  return folder;
153  }
154  }
155 
156  // Check in Visual Studio.app/
157  // In that case the name of the addin is used
158  var addinPath = IOPath.Combine(Path, $"Contents/Resources/lib/monodevelop/AddIns/{addinName}");
159  if (File.Exists(IOPath.Combine(addinPath, addinAssembly)))
160  return addinPath;
161  }
162 
163  return null;
164  }

166  private static string[] GetAnalyzers(string path)
167  {
168  var analyzersDirectory = IOPath.GetFullPath(IOPath.Combine(path, "Analyzers"));
169 
170  if (Directory.Exists(analyzersDirectory))

172 
173  return Array.Empty<string>();
174  }

176  public string[] GetAnalyzers()
177  {
178  var vstuPath = GetExtensionPath();
179  if (string.IsNullOrEmpty(vstuPath))
180  return Array.Empty<string>();
181 
182  if (VisualStudioEditor.IsOSX)
183  return GetAnalyzers(vstuPath);
184 
185  if (VisualStudioEditor.IsWindows)
186  {
187  var analyzers = GetAnalyzers(vstuPath);
188  if (analyzers?.Length > 0)
189  return analyzers;
190 
191  var bridge = GetWindowsBridgeFromRegistry();
192  if (File.Exists(bridge))
193  return GetAnalyzers(IOPath.Combine(IOPath.GetDirectoryName(bridge), ".."));
194  }
195 
196  // Local assets
197  // return FileUtility.FindPackageAssetFullPath("Analyzers a:packages", ".Analyzers.dll");
198  return Array.Empty<string>();
199  }
