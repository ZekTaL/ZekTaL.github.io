308  private static string[] CollectPathsFromToolbox(string toolboxRiderRootPath, string dirName, string searchPattern,
309  bool isMac)
310  {
311  if (!Directory.Exists(toolboxRiderRootPath))
312  return new string[0];
313 
314  var channelDirs = Directory.GetDirectories(toolboxRiderRootPath);
315  var paths = channelDirs.SelectMany(channelDir =>
316  {
317  try
318  {
319  // use history.json - last entry stands for the active build https://jetbrains.slack.com/archives/C07KNP99D/p1547807024066500?thread_ts=1547731708.057700&cid=C07KNP99D
320  var historyFile = Path.Combine(channelDir, ".history.json");
321  if (File.Exists(historyFile))
322  {
323  var json = File.ReadAllText(historyFile);
324  var build = ToolboxHistory.GetLatestBuildFromJson(json);
325  if (build != null)
326  {
327  var buildDir = Path.Combine(channelDir, build);
328  var executablePaths = GetExecutablePaths(dirName, searchPattern, isMac, buildDir);
329  if (executablePaths.Any())
330  return executablePaths;
331  }
332  }
333 
334  var channelFile = Path.Combine(channelDir, ".channel.settings.json");
335  if (File.Exists(channelFile))
336  {
337  var json = File.ReadAllText(channelFile).Replace("active-application", "active_application");
338  var build = ToolboxInstallData.GetLatestBuildFromJson(json);
339  if (build != null)
340  {
341  var buildDir = Path.Combine(channelDir, build);
342  var executablePaths = GetExecutablePaths(dirName, searchPattern, isMac, buildDir);
343  if (executablePaths.Any())
344  return executablePaths;
345  }
346  }
347 
348  // changes in toolbox json files format may brake the logic above, so return all found Rider installations
349  return Directory.GetDirectories(channelDir)
350  .SelectMany(buildDir => GetExecutablePaths(dirName, searchPattern, isMac, buildDir));
351  }
352  catch (Exception e)
353  {
354  // do not write to Debug.Log, just log it.
355  Logger.Warn($"Failed to get RiderPath from {channelDir}", e);
356  }
357 
358  return new string[0];
359  })
360  .Where(c => !string.IsNullOrEmpty(c))
361  .ToArray();
362  return paths;
363  }
