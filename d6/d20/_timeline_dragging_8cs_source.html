180  {

297  {
298  if (timeline == null)
299  return DragAndDropVisualMode.Rejected;
300 
301  // if we are over a target track, defer to track binding system (implemented in TrackGUIs), unless we are a groupTrack
302  if (targetTrack != null && (targetTrack as GroupTrack) == null)
303  return DragAndDropVisualMode.None;
304 
305  if (targetTrack != null && targetTrack.lockedInHierarchy)
306  return DragAndDropVisualMode.Rejected;
307 
308  var tracksWithBinding = objectsBeingDropped.SelectMany(TypeUtility.GetTracksCreatableFromObject).Distinct();
309  if (!tracksWithBinding.Any())
310  return DragAndDropVisualMode.None;
311 
312  if (perform)
313  {
314  System.Action<Type> onResolve = trackType =>
315  {
316  foreach (var obj in objectsBeingDropped)
317  {
318  if (!obj.IsPrefab() && TypeUtility.IsTrackCreatableFromObject(obj, trackType))
319  {

321  if (insertBefore != null)
322  {
323  if (targetTrack != null)
324  targetTrack.MoveLastTrackBefore(insertBefore);
325  else
326  timeline.MoveLastTrackBefore(insertBefore);
327  }
328 
329  TimelineHelpers.Bind(newTrack, obj, director);
330  }
331  }
332  TimelineEditor.Refresh(RefreshReason.ContentsAddedOrRemoved);
333  };
334  typeResolver(tracksWithBinding, onResolve, k_SelectTrackWithBinding);
335  }

340  public static DragAndDropVisualMode HandleClipPaneObjectDragAndDrop(IEnumerable<UnityObject> objectsBeingDropped, TrackAsset targetTrack, bool perform, TimelineAsset timeline, TrackAsset parent, PlayableDirector director, double candidateTime, TypeResolver typeResolver, TrackAsset insertBefore = null)
341  {
342  if (timeline == null)
343  return DragAndDropVisualMode.Rejected;
344 
345  // locked tracks always reject
346  if (targetTrack != null && targetTrack.lockedInHierarchy)
347  return DragAndDropVisualMode.Rejected;
348 
349  // treat group tracks as having no track
350  if (targetTrack is GroupTrack)
351  {
352  parent = targetTrack;
353  targetTrack = null;
354  }
355 
356  // Special case for monoscripts, since they describe the type
357  if (objectsBeingDropped.Any(o => o is MonoScript))
358  return HandleClipPaneMonoScriptDragAndDrop(objectsBeingDropped.OfType<MonoScript>(), targetTrack, perform, timeline, parent, director, candidateTime);
359 
360  // no unity objects, or explicit exceptions
361  if (!objectsBeingDropped.Any() || objectsBeingDropped.Any(o => !ValidateObjectDrop(o)))
362  return DragAndDropVisualMode.Rejected;
363 
364  // reject scene references if we have no context
365  if (director == null && objectsBeingDropped.Any(o => o.IsSceneObject()))
366  return DragAndDropVisualMode.Rejected;
367 
368  var validTrackTypes = objectsBeingDropped.SelectMany(o => TypeUtility.GetTrackTypesForObject(o)).Distinct().ToList();
369  // special case for playable assets
370  if (objectsBeingDropped.Any(o => TypeUtility.IsConcretePlayableAsset(o.GetType())))
371  {
372  var playableAssets = objectsBeingDropped.OfType<IPlayableAsset>().Where(o => TypeUtility.IsConcretePlayableAsset(o.GetType()));
373  return HandleClipPanePlayableAssetDragAndDrop(playableAssets, targetTrack, perform, timeline, parent, director, candidateTime, typeResolver);
374  }
375 
376  var markerTypes = objectsBeingDropped.SelectMany(o => TypeUtility.MarkerTypesWithFieldForObject(o)).Distinct();
377 
378  // No tracks or markers support this object
379  if (!(markerTypes.Any() || validTrackTypes.Any()))
380  {
381  return DragAndDropVisualMode.Rejected;
382  }
383  // track is not compatible with marker
384  if (targetTrack != null && markerTypes.Any(o => !TypeUtility.DoesTrackSupportMarkerType(targetTrack, o)))
385  {
386  // track is not compatible with object
387  if (!validTrackTypes.Contains(targetTrack.GetType()))
388  return DragAndDropVisualMode.Rejected;
389  }
390 
391  // there is no target track, dropping to empty space, or onto a group
392  if (perform)
393  {
394  // choose track and then clip
395  if (targetTrack == null)
396  {
397  var createdTrack = HandleTrackAndItemCreation(objectsBeingDropped, candidateTime, typeResolver, timeline, parent, validTrackTypes, insertBefore);
398  if (!createdTrack)
399  {
400  timeline.CreateMarkerTrack();
401  HandleItemCreation(objectsBeingDropped, timeline.markerTrack, candidateTime, typeResolver, true); // menu is always popped if ambiguous choice
402  }
403  }
404  // just choose clip/marker
405  else
406  {
407  HandleItemCreation(objectsBeingDropped, targetTrack, candidateTime, typeResolver, true); // menu is always popped if ambiguous choice
408  }
409  }
410 
411  return DragAndDropVisualMode.Copy;
412  }

551  bool validDrag = ValidDrag(targetItem, trackDragData.draggedItems);

557  if (draggedTracks.Count == 0)
558  return DragAndDropVisualMode.None;
559 
560  if (parentItem != null)

563  if (parentActor != null && parentActor.track != null)

566  return DragAndDropVisualMode.Rejected;

568  if (draggedTracks.Any(x => !TimelineCreateUtilities.ValidateParentTrack(parentActor.track, x.GetType())))
569  return DragAndDropVisualMode.Rejected;

577  }

580  {

582  if (groupGUI != null)
583  groupGUI.isDropTarget = true;
584  }
585 
586  if (perform)

589  var parentActor = parentItem as TimelineGroupGUI;
590

592  targetParent = parentActor.track;
593

611  public static void HandleBindingDragAndDrop(TrackAsset dropTarget, Type requiredBindingType)
612  {
613  var objectBeingDragged = DragAndDrop.objectReferences[0];
614 
615  var action = BindingUtility.GetBindingAction(requiredBindingType, objectBeingDragged);
616  DragAndDrop.visualMode = action == BindingAction.DoNotBind
617  ? DragAndDropVisualMode.Rejected
618  : DragAndDropVisualMode.Link;
619 
620  if (action == BindingAction.DoNotBind || Event.current.type != EventType.DragPerform)
621  return;
622 
623  var director = TimelineEditor.inspectedDirector;
624 
625  switch (action)
626  {
627  case BindingAction.BindDirectly:
628  {
629  BindingUtility.Bind(director, dropTarget, objectBeingDragged);
630  break;
631  }
632  case BindingAction.BindToExistingComponent:
633  {
634  var gameObjectBeingDragged = objectBeingDragged as GameObject;
635  Debug.Assert(gameObjectBeingDragged != null, "The object being dragged was detected as being a GameObject");
636 
637  BindingUtility.Bind(director, dropTarget, gameObjectBeingDragged.GetComponent(requiredBindingType));
638  break;
639  }
640  case BindingAction.BindToMissingComponent:
641  {
642  var gameObjectBeingDragged = objectBeingDragged as GameObject;
643  Debug.Assert(gameObjectBeingDragged != null, "The object being dragged was detected as being a GameObject");
644 
645  var typeNameOfComponent = requiredBindingType.ToString().Split(".".ToCharArray()).Last();
646  var bindMenu = new GenericMenu();
647 
648  bindMenu.AddItem(
649  EditorGUIUtility.TextContent("Create " + typeNameOfComponent + " on " + gameObjectBeingDragged.name),
650  false,
651  nullParam => BindingUtility.Bind(director, dropTarget, Undo.AddComponent(gameObjectBeingDragged, requiredBindingType)),
