533  public void EnsurePlayHeadIsVisible()
534  {
535  double minDisplayedTime = PixelToTime(timeAreaRect.xMin);
536  double maxDisplayedTime = PixelToTime(timeAreaRect.xMax);
537 
538  double currentTime = editSequence.time;
539  if (currentTime >= minDisplayedTime && currentTime <= maxDisplayedTime)
540  return;
541 
542  float displayedTimeRange = (float)(maxDisplayedTime - minDisplayedTime);
543  float minimumTimeToDisplay = (float)currentTime - displayedTimeRange / 2.0f;
544  float maximumTimeToDisplay = (float)currentTime + displayedTimeRange / 2.0f;
545  SetTimeAreaShownRange(minimumTimeToDisplay, maximumTimeToDisplay);
546  }
547 
548  public void SetPlayHeadToMiddle()
549  {
550  double minDisplayedTime = PixelToTime(timeAreaRect.xMin);
551  double maxDisplayedTime = PixelToTime(timeAreaRect.xMax);
552 
553  double currentTime = editSequence.time;
554  float displayedTimeRange = (float)(maxDisplayedTime - minDisplayedTime);
555 
556  if (currentTime >= minDisplayedTime && currentTime <= maxDisplayedTime)
557  {
558  if (currentTime < minDisplayedTime + displayedTimeRange / 2)
559  return;
560  }
561 
562  const float kCatchUpSpeed = 3f;
563  float realDelta = Mathf.Clamp(Time.realtimeSinceStartup - m_LastTime, 0f, 1f) * kCatchUpSpeed;
564  float scrollCatchupAmount = kCatchUpSpeed * realDelta * displayedTimeRange / 2;
565 
566  if (currentTime < minDisplayedTime)
567  {
568  SetTimeAreaShownRange((float)currentTime, (float)currentTime + displayedTimeRange);
569  }
570  else if (currentTime > maxDisplayedTime)
571  {
572  SetTimeAreaShownRange((float)currentTime - displayedTimeRange + scrollCatchupAmount, (float)currentTime + scrollCatchupAmount);
573  }
574  else if (currentTime > minDisplayedTime + displayedTimeRange / 2)
575  {
576  float targetMinDisplayedTime = Mathf.Min((float)minDisplayedTime + scrollCatchupAmount,
577  (float)(currentTime - displayedTimeRange / 2));
578  SetTimeAreaShownRange(targetMinDisplayedTime, targetMinDisplayedTime + displayedTimeRange);
579  }
580  }
