803  protected virtual void LateUpdate()
804  {
805  if (!m_Content)
806  return;
807 
808  EnsureLayoutHasRebuilt();
809  UpdateBounds();
810  float deltaTime = Time.unscaledDeltaTime;
811  Vector2 offset = CalculateOffset(Vector2.zero);
812  if (!m_Dragging && (offset != Vector2.zero || m_Velocity != Vector2.zero))
813  {
814  Vector2 position = m_Content.anchoredPosition;
815  for (int axis = 0; axis < 2; axis++)
816  {
817  // Apply spring physics if movement is elastic and content has an offset from the view.
818  if (m_MovementType == MovementType.Elastic && offset[axis] != 0)
819  {
820  float speed = m_Velocity[axis];
821  float smoothTime = m_Elasticity;
822  if (m_Scrolling)
823  smoothTime *= 3.0f;
824  position[axis] = Mathf.SmoothDamp(m_Content.anchoredPosition[axis], m_Content.anchoredPosition[axis] + offset[axis], ref speed, smoothTime, Mathf.Infinity, deltaTime);
825  if (Mathf.Abs(speed) < 1)
826  speed = 0;
827  m_Velocity[axis] = speed;
828  }
829  // Else move content according to velocity with deceleration applied.
830  else if (m_Inertia)
831  {
832  m_Velocity[axis] *= Mathf.Pow(m_DecelerationRate, deltaTime);
833  if (Mathf.Abs(m_Velocity[axis]) < 1)
834  m_Velocity[axis] = 0;
835  position[axis] += m_Velocity[axis] * deltaTime;
836  }
837  // If we have neither elaticity or friction, there shouldn't be any velocity.
838  else
839  {
840  m_Velocity[axis] = 0;
841  }
842  }
843 
844  if (m_MovementType == MovementType.Clamped)
845  {
846  offset = CalculateOffset(position - m_Content.anchoredPosition);
847  position += offset;
848  }
849 
850  SetContentAnchoredPosition(position);
851  }
852 
853  if (m_Dragging && m_Inertia)
854  {
855  Vector3 newVelocity = (m_Content.anchoredPosition - m_PrevPosition) / deltaTime;
856  m_Velocity = Vector3.Lerp(m_Velocity, newVelocity, deltaTime * 10);
857  }
858 
859  if (m_ViewBounds != m_PrevViewBounds || m_ContentBounds != m_PrevContentBounds || m_Content.anchoredPosition != m_PrevPosition)
860  {
861  UpdateScrollbars(offset);
862  UISystemProfilerApi.AddMarker("ScrollRect.value", this);
863  m_OnValueChanged.Invoke(normalizedPosition);
864  UpdatePrevData();
865  }
866  UpdateScrollbarVisibility();
867  m_Scrolling = false;
868  }

1033  // The new content localPosition, in the space of the view.

1294  internal static void AdjustBounds(ref Bounds viewBounds, ref Vector2 contentPivot, ref Vector3 contentSize, ref Vector3 contentPos)
1295  {
1296  // Make sure content bounds are at least as large as view by adding padding if not.
1297  // One might think at first that if the content is smaller than the view, scrolling should be allowed.
1298  // However, that's not how scroll views normally work.
1299  // Scrolling is *only* possible when content is *larger* than view.
1300  // We use the pivot of the content rect to decide in which directions the content bounds should be expanded.
1301  // E.g. if pivot is at top, bounds are expanded downwards.
1302  // This also works nicely when ContentSizeFitter is used on the content.
1303  Vector3 excess = viewBounds.size - contentSize;
1304  if (excess.x > 0)
1305  {
1306  contentPos.x -= excess.x * (contentPivot.x - 0.5f);
1307  contentSize.x = viewBounds.size.x;
1308  }
1309  if (excess.y > 0)
1310  {
1311  contentPos.y -= excess.y * (contentPivot.y - 0.5f);
1312  contentSize.y = viewBounds.size.y;
1313  }
1314  }
