38  static FullNameFilter OptimizedGroupFilter(string s)
39  {
40  if (s.Length >= 2)
41  {
42  // A common case is that the regex we are filtering by is of the form
43  // ^JUST_A_NAME$
44  // so we have a regex that matches _precisely_ one string. This can be done without a regex, which
45  // is much much faster.
46  if (s[0] == '^' && s[s.Length - 1] == '$')
47  {
48  var raw = s.Substring(1, s.Length - 2);
49  var escaped = Regex.Escape(raw);
50  // if a regex is the same when we escape it, it means that it doesn't contain any characters
51  // with any meaning in the regex. Hence the regex is just a plain name.
52  if (raw.Equals(escaped, StringComparison.Ordinal))
53  return new FullNameFilter(raw);
54  }
55  }
56 
57  return new FullNameFilter(s) { IsRegex = true };
58  }

60  private static void AddFilters(List<ITestFilter> filters, string[] values, Func<string, TestFilter> builder)
61  {
62  if (values == null || values.Length == 0)

64  return;

66 
67  var inclusionFilters = values.Where(v => !v.StartsWith("!")).Select(v => builder(v) as ITestFilter).ToArray();
68  var exclusionFilters = values.Where(v => v.StartsWith("!"))
69  .Select(v => new NotFilter(builder(v.Substring(1))) as ITestFilter)
70  .ToArray();
71  if (inclusionFilters.Length > 0 && exclusionFilters.Length > 0)
72  {
73  filters.Add(new AndFilter(new OrFilter(inclusionFilters), new AndFilter(exclusionFilters)));
74  }
75  else if (inclusionFilters.Length > 0)
