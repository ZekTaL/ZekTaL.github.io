835  internal virtual Playable OnCreateClipPlayableGraph(PlayableGraph graph, GameObject go, IntervalTree<RuntimeElement> tree)
836  {
837  if (tree == null)
838  throw new ArgumentException("IntervalTree argument cannot be null", "tree");
839 
840  if (go == null)
841  throw new ArgumentException("GameObject argument cannot be null", "go");
842 
843  s_BuildData.Clear();
844  GatherCompilableTracks(s_BuildData.trackList);
845 
846  // nothing to compile
847  if (s_BuildData.trackList.Count == 0)
848  return Playable.Null;
849 
850  // check if layers are supported
851  Playable layerMixer = Playable.Null;
852  ILayerable layerable = this as ILayerable;
853  if (layerable != null)
854  layerMixer = layerable.CreateLayerMixer(graph, go, s_BuildData.trackList.Count);
855 
856  if (layerMixer.IsValid())
857  {
858  for (int i = 0; i < s_BuildData.trackList.Count; i++)
859  {
860  var mixer = s_BuildData.trackList[i].CompileClips(graph, go, s_BuildData.trackList[i].clips, tree);
861  if (mixer.IsValid())
862  {
863  graph.Connect(mixer, 0, layerMixer, i);
864  layerMixer.SetInputWeight(i, 1.0f);
865  }
866  }
867  return layerMixer;
868  }
869 
870  // one track compiles. Add track mixer and clips
871  if (s_BuildData.trackList.Count == 1)
872  return s_BuildData.trackList[0].CompileClips(graph, go, s_BuildData.trackList[0].clips, tree);
873 
874  // no layer mixer provided. merge down all clips.
875  for (int i = 0; i < s_BuildData.trackList.Count; i++)
876  s_BuildData.clipList.AddRange(s_BuildData.trackList[i].clips);
877 
878 #if UNITY_EDITOR
879  bool applyWarning = false;
880  for (int i = 0; i < s_BuildData.trackList.Count; i++)
881  applyWarning |= i > 0 && s_BuildData.trackList[i].hasCurves;
882 
883  if (applyWarning)
884  Debug.LogWarning("A layered track contains animated fields, but no layer mixer has been provided. Animated fields on layers will be ignored. Override CreateLayerMixer in " + s_BuildData.trackList[0].GetType().Name + " and return a valid playable to support animated fields on layered tracks.");
885 #endif
886  // compile all the clips into a single mixer
887  return CompileClips(graph, go, s_BuildData.clipList, tree);
