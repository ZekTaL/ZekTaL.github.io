246  static void DrawBlendLine(Rect rect, BlendAngle blendAngle, float width, Color color)
247  {
248  var halfWidth = width / 2.0f;
249  Vector2 p0, p1;
250  var inverse = 1.0f;
251  if (blendAngle == BlendAngle.Descending)
252  {
253  p0 = rect.min;
254  p1 = rect.max;
255  }
256  else
257  {
258  p0 = new Vector2(rect.xMax, rect.yMin);
259  p1 = new Vector2(rect.xMin, rect.yMax);
260  inverse = -1.0f;
261  }
262  s_BlendLines[0] = new Vector3(p0.x - halfWidth, p0.y + halfWidth * inverse);
263  s_BlendLines[1] = new Vector3(p1.x - halfWidth, p1.y + halfWidth * inverse);
264  s_BlendLines[2] = new Vector3(p1.x + halfWidth, p1.y - halfWidth * inverse);
265  s_BlendLines[3] = new Vector3(p0.x + halfWidth, p0.y - halfWidth * inverse);
266  Graphics.DrawPolygonAA(color, s_BlendLines);
