600  void CalculateLoopRects(Rect trackRect, WindowState state)
601  {
602  if (!m_ClipViewDirty)
603  return;
604 
605  m_LoopRects.Clear();
606  if (clip.duration < WindowState.kTimeEpsilon)
607  return;
608 
609  var times = TimelineHelpers.GetLoopTimes(clip);
610  var loopDuration = TimelineHelpers.GetLoopDuration(clip);
611  m_MinLoopIndex = -1;
612 
613  // we have a hold, no need to compute all loops
614  if (!supportsLooping)
615  {
616  if (times.Length > 1)
617  {
618  var t = times[1];
619  float loopTime = (float)(clip.duration - t);
620  m_LoopRects.Add(ProjectRectOnTimeline(new Rect((float)(t + clip.start), 0, loopTime, 0), trackRect, state));
621  }
622  return;
623  }
624 
625  var range = state.timeAreaShownRange;
626  var visibleStartTime = range.x - clip.start;
627  var visibleEndTime = range.y - clip.start;
628 
629  for (int i = 1; i < times.Length; i++)
630  {
631  var t = times[i];
632 
633  // don't draw off screen loops
634  if (t > visibleEndTime)
635  break;
636 
637  float loopTime = Mathf.Min((float)(clip.duration - t), (float)loopDuration);
638  var loopEnd = t + loopTime;
639 
640  if (loopEnd < visibleStartTime)
641  continue;
642 
643  m_LoopRects.Add(ProjectRectOnTimeline(new Rect((float)(t + clip.start), 0, loopTime, 0), trackRect, state));
644 
645  if (m_MinLoopIndex == -1)
646  m_MinLoopIndex = i;
647  }
648  }

663  public IEnumerable<Edge> SnappableEdgesFor(IAttractable attractable, ManipulateEdges manipulateEdges)
664  {
665  var edges = new List<Edge>();
666 
667  bool canAddEdges = !parent.muted;
668 
669  if (canAddEdges)
670  {
671  // Hack: Trim Start in Ripple mode should not have any snap point added
672  if (EditMode.editType == EditMode.EditType.Ripple && manipulateEdges == ManipulateEdges.Left)
673  return edges;
674 
675  if (attractable != this)
676  {
677  if (EditMode.editType == EditMode.EditType.Ripple)
678  {
679  bool skip = false;
680 
681  // Hack: Since Trim End and Move in Ripple mode causes other snap point to move on the same track (which is not supported), disable snapping for this special cases...
682  // TODO Find a proper way to have different snap edges for each edit mode.
683  if (manipulateEdges == ManipulateEdges.Right)
684  {
685  var otherClipGUI = attractable as TimelineClipGUI;
686  skip = otherClipGUI != null && otherClipGUI.parent == parent;
687  }
688  else if (manipulateEdges == ManipulateEdges.Both)
689  {
690  var moveHandler = attractable as MoveItemHandler;
691  skip = moveHandler != null && moveHandler.movingItems.Any(clips => clips.targetTrack == clip.parentTrack && clip.start >= clips.start);
692  }
693 
694  if (skip)
695  return edges;
696  }
697 
698  AddEdge(edges, clip.start);
699  AddEdge(edges, clip.end);
700  }
701  else
702  {
703  if (manipulateEdges == ManipulateEdges.Right)
704  {
705  var d = TimelineHelpers.GetClipAssetEndTime(clip);
706 
707  if (d < double.MaxValue)
708  {
709  if (clip.SupportsLooping())
710  {
711  var l = TimelineHelpers.GetLoopDuration(clip);
712 
713  var shownTime = TimelineWindow.instance.state.timeAreaShownRange;
714  do
715  {
716  AddEdge(edges, d, false);
717  d += l;
