746  public Selectable FindSelectable(Vector3 dir)
747  {
748  dir = dir.normalized;
749  Vector3 localDir = Quaternion.Inverse(transform.rotation) * dir;
750  Vector3 pos = transform.TransformPoint(GetPointOnRectEdge(transform as RectTransform, localDir));
751  float maxScore = Mathf.NegativeInfinity;
752  float maxFurthestScore = Mathf.NegativeInfinity;
753  float score = 0;
754 
755  bool wantsWrapAround = navigation.wrapAround && (m_Navigation.mode == Navigation.Mode.Vertical || m_Navigation.mode == Navigation.Mode.Horizontal);
756 
757  Selectable bestPick = null;
758  Selectable bestFurthestPick = null;
759 
760  for (int i = 0; i < s_SelectableCount; ++i)
761  {
762  Selectable sel = s_Selectables[i];
763 
764  if (sel == this)
765  continue;
766 
767  if (!sel.IsInteractable() || sel.navigation.mode == Navigation.Mode.None)
768  continue;
769 
770 #if UNITY_EDITOR
771  // Apart from runtime use, FindSelectable is used by custom editors to
772  // draw arrows between different selectables. For scene view cameras,
773  // only selectables in the same stage should be considered.
774  if (Camera.current != null && !UnityEditor.SceneManagement.StageUtility.IsGameObjectRenderedByCamera(sel.gameObject, Camera.current))
775  continue;
776 #endif
777 
778  var selRect = sel.transform as RectTransform;
779  Vector3 selCenter = selRect != null ? (Vector3)selRect.rect.center : Vector3.zero;
780  Vector3 myVector = sel.transform.TransformPoint(selCenter) - pos;
781 
782  // Value that is the distance out along the direction.
783  float dot = Vector3.Dot(dir, myVector);
784 
785  // If element is in wrong direction and we have wrapAround enabled check and cache it if furthest away.
786  if (wantsWrapAround && dot < 0)
787  {
788  score = -dot * myVector.sqrMagnitude;
789 
790  if (score > maxFurthestScore)
791  {
792  maxFurthestScore = score;
793  bestFurthestPick = sel;
794  }
795 
796  continue;
797  }
798 
799  // Skip elements that are in the wrong direction or which have zero distance.
800  // This also ensures that the scoring formula below will not have a division by zero error.
801  if (dot <= 0)
802  continue;
803 
804  // This scoring function has two priorities:
805  // - Score higher for positions that are closer.
806  // - Score higher for positions that are located in the right direction.
807  // This scoring function combines both of these criteria.
808  // It can be seen as this:
809  // Dot (dir, myVector.normalized) / myVector.magnitude
810  // The first part equals 1 if the direction of myVector is the same as dir, and 0 if it's orthogonal.
811  // The second part scores lower the greater the distance is by dividing by the distance.
812  // The formula below is equivalent but more optimized.
813  //
814  // If a given score is chosen, the positions that evaluate to that score will form a circle
815  // that touches pos and whose center is located along dir. A way to visualize the resulting functionality is this:
816  // From the position pos, blow up a circular balloon so it grows in the direction of dir.
817  // The first Selectable whose center the circular balloon touches is the one that's chosen.
818  score = dot / myVector.sqrMagnitude;
819 
820  if (score > maxScore)
821  {
822  maxScore = score;
823  bestPick = sel;
824  }
825  }
826 
827  if (wantsWrapAround && null == bestPick) return bestFurthestPick;
828 
829  return bestPick;
830  }
