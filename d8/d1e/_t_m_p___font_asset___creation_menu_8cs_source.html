152  {

154 
155  if (targets == null)
156  {
157  Debug.LogWarning("A Font file must first be selected in order to create a Font Asset.");
158  return;
159  }
160 
161  for (int i = 0; i < targets.Length; i++)
162  {
163  Object target = targets[i];
164 
165  // Make sure the selection is a font file
166  if (target == null || target.GetType() != typeof(Font))

168  Debug.LogWarning("Selected Object [" + target.name + "] is not a Font file. A Font file must be selected in order to create a Font Asset.", target);

171 
172  CreateFontAssetFromSelectedObject(target);
173  }
