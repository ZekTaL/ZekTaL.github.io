322  static void ScanProjectFile(AssetFileRecord fileRecord)
323  {
324  if (m_CancelScanProcess)
325  return;
326 
327  // Read the asset data file
328  string assetDataFile;
329  bool hasDataFileChanged = false;
330 
331  try
332  {
333  assetDataFile = File.ReadAllText(m_ProjectPath + "/" + fileRecord.assetFilePath);
334  }
335  catch
336  {
337  // Continue to the next asset if we can't read the current one.
338  return;
339  }
340 
341  // Check if asset file references any text components.
342  if (assetDataFile.Contains(k_TextMeshProScriptID) || assetDataFile.Contains(k_TextMeshProUGUIScriptID))
343  {
344  float characterSpacingValue = 0;
345  float newCharacterSpacingValue = 0;
346  float wordSpacingValue = 0;
347  float newWordSpacingValue = 0;
348  float lineSpacingValue = 0;
349  float newLineSpacingValue = 0;
350  float paragraphSpacingValue = 0;
351  float newParagraphSpacingValue = 0;
352 
353  float fontSize = 0;
354  float samplingPointSize = 0;
355  float faceScale = 1;
356 
357  List<string> lines = assetDataFile.Split('\n').ToList();
358  int serializedVersionInsertionIndex = 0;
359 
360  int readingFlag = 0;
361 
362  // Read through each lines of the asset file
363  for (int i = 0; i < lines.Count; i++)
364  {
365  string line = lines[i];
366 
367  // Track potential line index to insert serializedVersion property
368  if (line.Contains("MonoBehaviour:"))
369  {
370  serializedVersionInsertionIndex = i + 1;
371  continue;
372  }
373 
374  // Read until we find the line that contains a reference to a text component
375  if (readingFlag == 0 && (line.Contains(k_TextMeshProScriptID) || line.Contains(k_TextMeshProUGUIScriptID)))
376  {
377  // Check if spacing values for this component have already been converted
378  if (lines[serializedVersionInsertionIndex].Contains(" m_SerializedVersion: 1"))
379  {
380  readingFlag = 0;
381  continue;
382  }
383 
384  lines.Insert(serializedVersionInsertionIndex, " m_SerializedVersion: 1");
385  readingFlag = 1;
386  continue;
387  }
388 
389  // Keep reading until we find the font asset property field.
390  if (readingFlag == 1)
391  {
392  // Check for font asset property
393  if (line.Contains(k_FontAssetProperty))
394  {
395  int guidIndex = line.IndexOf("guid: ", StringComparison.InvariantCulture);
396  if (guidIndex != -1)
397  {
398  string guid = line.Substring(guidIndex + 6, 32);
399  TMP_FontAsset fontAsset = AssetDatabase.LoadAssetAtPath<TMP_FontAsset>(AssetDatabase.GUIDToAssetPath(guid));
400  if (fontAsset != null)
401  {
402  samplingPointSize = fontAsset.faceInfo.pointSize;
403  faceScale = fontAsset.faceInfo.scale;
404  }
405  }
406 
407  readingFlag = 2;
408  continue;
409  }
410  }
411 
412  // Read font size property
413  if (readingFlag == 2)
414  {
415  if (line.Contains(k_FontSizeProperty))
416  {
417  fontSize = float.Parse(line.Split(':')[1], NumberStyles.Float, CultureInfo.InvariantCulture);
418  readingFlag = 3;
419  continue;
420  }
421  }
422 
423  // Check for the spacing properties that need to be converted
424  if (readingFlag == 3)
425  {
426  // Read character spacing
427  if (line.Contains(k_CharacterSpacingProperty))
428  {
429  characterSpacingValue = float.Parse(line.Split(':')[1], NumberStyles.Float, CultureInfo.InvariantCulture);
430  if (characterSpacingValue != 0)
431  {
432  // Convert character spacing value.
433  newCharacterSpacingValue = characterSpacingValue * faceScale / (samplingPointSize * 0.01f);
434  lines[i] = lines[i].Replace(k_CharacterSpacingProperty + characterSpacingValue, k_CharacterSpacingProperty + newCharacterSpacingValue);
435 
436  hasDataFileChanged = true;
437  }
438  continue;
439  }
440 
441  // Read word spacing
442  if (line.Contains(k_WordSpacingProperty))
443  {
444  // Get the character spacing value
445  wordSpacingValue = float.Parse(line.Split(':')[1], NumberStyles.Float, CultureInfo.InvariantCulture);
446  if (wordSpacingValue != 0)
447  {
448  // Convert character spacing value.
449  newWordSpacingValue = wordSpacingValue * faceScale / (samplingPointSize * 0.01f);
450  lines[i] = lines[i].Replace(k_WordSpacingProperty + wordSpacingValue, k_WordSpacingProperty + newWordSpacingValue);
451 
452  hasDataFileChanged = true;
453  }
454  continue;
455  }
456 
457  // Read line spacing
458  if (line.Contains(k_LineSpacingProperty))
459  {
460  // Get the value of line spacing value
461  lineSpacingValue = float.Parse(line.Split(':')[1], NumberStyles.Float, CultureInfo.InvariantCulture);
462  if (lineSpacingValue != 0)
463  {
464  // Convert line spacing value.
465  newLineSpacingValue = lineSpacingValue / (fontSize * 0.01f) * fontSize / samplingPointSize * faceScale;
466  lines[i] = lines[i].Replace(k_LineSpacingProperty + lineSpacingValue, k_LineSpacingProperty + newLineSpacingValue);
467 
468  hasDataFileChanged = true;
469  }
470  continue;
471  }
472 
473  // Read paragraph spacing
474  if (line.Contains(k_ParagraphSpacingProperty))
475  {
476  // Get the value of line spacing value
477  paragraphSpacingValue = float.Parse(line.Split(':')[1], NumberStyles.Float, CultureInfo.InvariantCulture);
478  if (paragraphSpacingValue != 0)
479  {
480  // Convert line spacing value.
481  newParagraphSpacingValue = paragraphSpacingValue / (fontSize * 0.01f) * fontSize / samplingPointSize * faceScale;
482  lines[i] = lines[i].Replace(k_ParagraphSpacingProperty + paragraphSpacingValue, k_ParagraphSpacingProperty + newParagraphSpacingValue);
483 
484  hasDataFileChanged = true;
485  }
486 
487  readingFlag = 4;
488  continue;
489  }
490  }
491 
492  // Done reading text component serialized data.
493  if (readingFlag == 4 && line.Contains("---"))
494  {
495  readingFlag = 0;
496 
497  string characterSpacingFormat = $"{(characterSpacingValue == 0 ? " " : $"{characterSpacingValue,10:F}{newCharacterSpacingValue,10:F}")}";
498  string wordSpacingFormat = $"{(wordSpacingValue == 0 ? " " : $"{wordSpacingValue,10:F}{newWordSpacingValue,10:F}")}";
499  string lineSpacingFormat = $"{(lineSpacingValue == 0 ? " " : $"{lineSpacingValue,10:F}{newLineSpacingValue,10:F}")}";
500  string paragraphSpacingFormat = $"{(paragraphSpacingValue == 0 ? " " : $"{paragraphSpacingValue,10:F}{newParagraphSpacingValue,10:F}")}";
501 
502  if (characterSpacingValue != 0 || lineSpacingValue != 0)
503  m_ProjectScanResults += $"{fileRecord.assetFilePath,-100}" + characterSpacingFormat + wordSpacingFormat + lineSpacingFormat + paragraphSpacingFormat + "\n";
504 
505  // Update asset data file
506  assetDataFile = string.Join("\n", lines);
507 
508  newCharacterSpacingValue = 0;
509  newWordSpacingValue = 0;
510  newLineSpacingValue = 0;
511  newParagraphSpacingValue = 0;
512  }
513  }
514  }
515 
516  // Check if asset file is a font asset
517  // if (assetDataFile.Contains(k_FontAssetScriptID))
518  // {
519  // float samplingPointSize;
520  // float normalSpacing;
521  // float newNormalSpacing;
522  // float boldSpacing;
523  // float newBoldSpacing;
524  // }
525 
526  if (hasDataFileChanged)
527  {
528  AssetModificationRecord modifiedAsset;
529  modifiedAsset.assetFilePath = fileRecord.assetFilePath;
530  modifiedAsset.assetDataFile = assetDataFile;
531 
532  m_ModifiedAssetList.Add(modifiedAsset);
533  }

811  private IEnumerator ScanProjectFiles()
812  {
813  m_IsAlreadyScanningProject = true;
814  string packageFullPath = EditorUtilities.TMP_EditorUtility.packageFullPath;
815 
816  // List containing assets that have been modified.
817  m_ProjectScanResults = k_ProjectScanReportDefaultText;
818  m_ModifiedAssetList.Clear();
819  m_ProgressPercentage = 0;
820 
821  // Read Conversion Data from Json file.
822  if (m_ConversionData == null)
823  m_ConversionData = JsonUtility.FromJson<AssetConversionData>(File.ReadAllText(packageFullPath + "/PackageConversionData.json"));
824 
825  // Get list of GUIDs for assets that might contain references to previous GUIDs that require updating.
826  string searchFolder = string.IsNullOrEmpty(m_ProjectFolderToScan) ? "Assets" : ("Assets/" + m_ProjectFolderToScan);
827  string[] guids = AssetDatabase.FindAssets("t:Object", new string[] { searchFolder }).Distinct().ToArray();
828 
829  k_ProjectScanLabelPrefix = "<b>Phase 1 - Filtering:</b> ";
830  m_ScanningTotalFiles = guids.Length;
831  m_ScanningCurrentFileIndex = 0;
832 
833  List<AssetFileRecord> projectFilesToScan = new List<AssetFileRecord>();
834 
835  foreach (var guid in guids)
836  {
837  if (m_CancelScanProcess)
838  break;
839 
840  string assetFilePath = AssetDatabase.GUIDToAssetPath(guid);
841 
842  m_ScanningCurrentFileIndex += 1;
843  m_ScanningCurrentFileName = assetFilePath;
844  m_ProgressPercentage = (float)m_ScanningCurrentFileIndex / m_ScanningTotalFiles;
845 
846  // Filter out file types we have no interest in searching
847  if (ShouldIgnoreFile(assetFilePath))
848  continue;
849 
850  string assetMetaFilePath = AssetDatabase.GetTextMetaFilePathFromAssetPath(assetFilePath);
851 
852  projectFilesToScan.Add(new AssetFileRecord(assetFilePath, assetMetaFilePath));
853 
854  yield return null;
855  }
856 
857  m_RemainingFilesToScan = m_ScanningTotalFiles = projectFilesToScan.Count;
858 
859  k_ProjectScanLabelPrefix = "<b>Phase 2 - Scanning:</b> ";
860 
861  for (int i = 0; i < m_ScanningTotalFiles; i++)
862  {
863  if (m_CancelScanProcess)
864  break;
865 
866  AssetFileRecord fileRecord = projectFilesToScan[i];
867 
868  ThreadPool.QueueUserWorkItem(Task =>
869  {
870  ScanProjectFileAsync(fileRecord);
871 
872  m_ScanningCurrentFileName = fileRecord.assetFilePath;
873 
874  int completedScans = m_ScanningTotalFiles - Interlocked.Decrement(ref m_RemainingFilesToScan);
875 
876  m_ScanningCurrentFileIndex = completedScans;
877  m_ProgressPercentage = (float)completedScans / m_ScanningTotalFiles;
878  });
879 
880  if (i % 64 == 0)
881  yield return new WaitForSeconds(2.0f);
882 
883  }
884 
885  while (m_RemainingFilesToScan > 0 && !m_CancelScanProcess)
886  yield return null;
887 
888  m_IsAlreadyScanningProject = false;
889  m_ScanningCurrentFileName = string.Empty;
890  }

893  static void ScanProjectFileAsync(AssetFileRecord fileRecord)
894  {
895  if (m_CancelScanProcess)
896  return;
897 
898  // Read the asset data file
899  string assetDataFile = string.Empty;
900  bool hasFileChanged = false;
901 
902  try
903  {
904  assetDataFile = File.ReadAllText(m_ProjectPath + "/" + fileRecord.assetFilePath);
905  }
906  catch
907  {
908  // Continue to the next asset if we can't read the current one.
909  return;
910  }
911 
912  // Read the asset meta data file
913  string assetMetaFile = File.ReadAllText(m_ProjectPath + "/" + fileRecord.assetMetaFilePath);
914  bool hasMetaFileChanges = false;
915 
916  foreach (AssetConversionRecord record in m_ConversionData.assetRecords)
917  {
918  if (assetDataFile.Contains(record.target))
919  {
920  hasFileChanged = true;
921 
922  assetDataFile = assetDataFile.Replace(record.target, record.replacement);
923  }
924 
926  if (assetMetaFile.Contains(record.target))
927  {
928  hasMetaFileChanges = true;
929 
930  assetMetaFile = assetMetaFile.Replace(record.target, record.replacement);
931  }
932  }
933 
934  if (hasFileChanged)
935  {
936  AssetModificationRecord modifiedAsset;
937  modifiedAsset.assetFilePath = fileRecord.assetFilePath;
938  modifiedAsset.assetDataFile = assetDataFile;
939 
940  m_ModifiedAssetList.Add(modifiedAsset);
941 
942  m_ProjectScanResults += fileRecord.assetFilePath + "\n";
943  }
944 
945  if (hasMetaFileChanges)
946  {
947  AssetModificationRecord modifiedAsset;
948  modifiedAsset.assetFilePath = fileRecord.assetMetaFilePath;
949  modifiedAsset.assetDataFile = assetMetaFile;
950 
951  m_ModifiedAssetList.Add(modifiedAsset);
952 
953  m_ProjectScanResults += fileRecord.assetMetaFilePath + "\n";
954  }
