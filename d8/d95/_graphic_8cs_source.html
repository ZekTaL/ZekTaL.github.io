813  public virtual bool Raycast(Vector2 sp, Camera eventCamera)
814  {
815  if (!isActiveAndEnabled)
816  return false;
817 
818  var t = transform;
819  var components = ListPool<Component>.Get();
820 
821  bool ignoreParentGroups = false;
822  bool continueTraversal = true;
823 
824  while (t != null)
825  {
826  t.GetComponents(components);
827  for (var i = 0; i < components.Count; i++)
828  {
829  var canvas = components[i] as Canvas;
830  if (canvas != null && canvas.overrideSorting)
831  continueTraversal = false;
832 
833  var filter = components[i] as ICanvasRaycastFilter;
834 
835  if (filter == null)
836  continue;
837 
838  var raycastValid = true;
839 
840  var group = components[i] as CanvasGroup;
841  if (group != null)
842  {
843  if (ignoreParentGroups == false && group.ignoreParentGroups)
844  {
845  ignoreParentGroups = true;
846  raycastValid = filter.IsRaycastLocationValid(sp, eventCamera);
847  }
848  else if (!ignoreParentGroups)
849  raycastValid = filter.IsRaycastLocationValid(sp, eventCamera);
850  }
851  else
852  {
853  raycastValid = filter.IsRaycastLocationValid(sp, eventCamera);
854  }
855 
856  if (!raycastValid)
857  {
858  ListPool<Component>.Release(components);
859  return false;
860  }
861  }
862  t = continueTraversal ? t.parent : null;
863  }
864  ListPool<Component>.Release(components);
865  return true;
866  }
