47  {

87  internal static void UpdateClipIntersection(TimelineClip blendOutClip, TimelineClip blendInClip)
88  {
89  if (!blendOutClip.SupportsBlending() || !blendInClip.SupportsBlending())
90  return;
91 
92  if (blendInClip.end < blendOutClip.end)
93  return;
94 
95  double duration = Math.Max(0, blendOutClip.start + blendOutClip.duration - blendInClip.start);
96  duration = duration <= kMinOverlapTime ? 0 : duration;
97  blendOutClip.blendOutDuration = duration;
98  blendInClip.blendInDuration = duration;
99 
100  var blendInMode = blendInClip.blendInCurveMode;
101  var blendOutMode = blendOutClip.blendOutCurveMode;

103  if (blendInMode == TimelineClip.BlendCurveMode.Manual && blendOutMode == TimelineClip.BlendCurveMode.Auto)
104  {
105  blendOutClip.mixOutCurve = CurveEditUtility.CreateMatchingCurve(blendInClip.mixInCurve);

144  internal static TrackAsset Duplicate(this TrackAsset track, IExposedPropertyTable sourceTable, IExposedPropertyTable destTable,
145  TimelineAsset destinationTimeline = null)
146  {
147  if (track == null)
148  return null;
149 
150  // if the destination is us, clear to avoid bad parenting (case 919421)
151  if (destinationTimeline == track.timelineAsset)
152  destinationTimeline = null;
153 
154  var timelineParent = track.parent as TimelineAsset;
155  var trackParent = track.parent as TrackAsset;
156  if (timelineParent == null && trackParent == null)
157  {
158  Debug.LogWarning("Cannot duplicate track because it is not parented to known type");
159  return null;
160  }
161 
162  // Determine who the final parent is. If we are pasting into another track, it's always the timeline.
163  // Otherwise it's the original parent
164  PlayableAsset finalParent = destinationTimeline != null ? destinationTimeline : track.parent;
165 
166  // grab the list of tracks to generate a name from (923360) to get the list of names
167  // no need to do this part recursively
168  var finalTrackParent = finalParent as TrackAsset;
169  var finalTimelineAsset = finalParent as TimelineAsset;
170  var otherTracks = (finalTimelineAsset != null) ? finalTimelineAsset.trackObjects : finalTrackParent.subTracksObjects;
171 
172  // Important to create the new objects before pushing the original undo, or redo breaks the
173  // sequence
174  var newTrack = TimelineHelpers.Clone(finalParent, track, sourceTable, destTable, finalParent);
175  newTrack.name = TimelineCreateUtilities.GenerateUniqueActorName(otherTracks, newTrack.name);
176 
177  RecursiveSubtrackClone(track, newTrack, sourceTable, destTable, finalParent);
178  TimelineCreateUtilities.SaveAssetIntoObject(newTrack, finalParent);
179  TimelineUndo.RegisterCreatedObjectUndo(newTrack, "Duplicate");
180  UndoExtensions.RegisterPlayableAsset(finalParent, "Duplicate");
181 
182  if (destinationTimeline != null) // other timeline
183  destinationTimeline.AddTrackInternal(newTrack);
184  else if (timelineParent != null) // this timeline, no parent
185  ReparentTracks(new List<TrackAsset> { newTrack }, timelineParent, timelineParent.GetRootTracks().Last(), false);
186  else // this timeline, with parent
187  trackParent.AddChild(newTrack);
188 
189  // Call the custom editor. this check prevents the call when copying to the clipboard
190  if (destinationTimeline == null || destinationTimeline == TimelineEditor.inspectedAsset)
191  {
192  var customEditor = CustomTimelineEditorCache.GetTrackEditor(newTrack);
193  try
194  {
195  customEditor.OnCreate(newTrack, track);
196  }
197  catch (Exception e)
198  {
199  Debug.LogException(e);
200  }
201  }
202 
203  return newTrack;

206  // Reparents a list of tracks to a new parent
207  // the new parent cannot be null (has to be track asset or sequence)
208  // the insertAfter can be null (will not reorder)
209  internal static bool ReparentTracks(List<TrackAsset> tracksToMove, PlayableAsset targetParent,
210  TrackAsset insertMarker = null, bool insertBefore = false)
211  {
212  var targetParentTrack = targetParent as TrackAsset;
213  var targetSequenceTrack = targetParent as TimelineAsset;
214 
215  if (tracksToMove == null || tracksToMove.Count == 0 || (targetParentTrack == null && targetSequenceTrack == null))
216  return false;
217 
218  // invalid parent type on a track
219  if (targetParentTrack != null && tracksToMove.Any(x => !TimelineCreateUtilities.ValidateParentTrack(targetParentTrack, x.GetType())))
220  return false;
221 
222  // no valid tracks means this is simply a rearrangement
223  var validTracks = tracksToMove.Where(x => x.parent != targetParent).ToList();
224  if (insertMarker == null && !validTracks.Any())
225  return false;
226 
227  var parents = validTracks.Select(x => x.parent).Where(x => x != null).Distinct().ToList();
228  // push the current state of the tracks that will change
229  foreach (var p in parents)
230  {
231  UndoExtensions.RegisterPlayableAsset(p, "Reparent");
232  }
233  UndoExtensions.RegisterTracks(validTracks, "Reparent");
234  UndoExtensions.RegisterPlayableAsset(targetParent, "Reparent");
235 
236  // need to reparent tracks first, before moving them.
237  foreach (var t in validTracks)
238  {
239  if (t.parent != targetParent)
240  {
241  TrackAsset toMoveParent = t.parent as TrackAsset;
242  TimelineAsset toMoveTimeline = t.parent as TimelineAsset;
243  if (toMoveTimeline != null)
244  {
245  toMoveTimeline.RemoveTrack(t);
246  }
247  else if (toMoveParent != null)
248  {
249  toMoveParent.RemoveSubTrack(t);
250  }
251 
252  if (targetParentTrack != null)
253  {
254  targetParentTrack.AddChild(t);
255  targetParentTrack.SetCollapsed(false);
256  }
257  else
258  {
259  targetSequenceTrack.AddTrackInternal(t);
260  }
261  }
262  }
263 
264 
265  if (insertMarker != null)
266  {
267  // re-ordering track. This is using internal APIs, so invalidation of the tracks must be done manually to avoid
268  // cache mismatches
269  var children = targetParentTrack != null ? targetParentTrack.subTracksObjects : targetSequenceTrack.trackObjects;
270  TimelineUtility.ReorderTracks(children, tracksToMove, insertMarker, insertBefore);
271  if (targetParentTrack != null)
272  targetParentTrack.Invalidate();
273  if (insertMarker.timelineAsset != null)
274  insertMarker.timelineAsset.Invalidate();
275  }
276 
277  return true;
278  }

403  // return true if there is a gap, false if there is an intersection

405  internal static bool GetGapAtTime(this TrackAsset track, double time, out double startGap, out double endGap)
406  {
407  startGap = 0;
408  endGap = Double.PositiveInfinity;
409 
410  if (track == null || !track.GetClips().Any())
411  {
412  return false;
413  }
414 
415  var discreteTime = new DiscreteTime(time);
416 
417  track.SortClips();
418  var sortedByStartTime = track.clips;
419  for (int i = 0; i < sortedByStartTime.Length; i++)
420  {
421  var clip = sortedByStartTime[i];
422 
423  // intersection
424  if (discreteTime >= new DiscreteTime(clip.start) && discreteTime < new DiscreteTime(clip.end))
425  {
426  endGap = time;
427  startGap = time;
428  return false;
429  }
430 
431  if (clip.end < time)
432  {
433  startGap = clip.end;
434  }
435  if (clip.start > time)
436  {
437  endGap = clip.start;
438  break;
439  }
440  }
441 
442  if (endGap - startGap < TimelineClip.kMinDuration)
443  {
444  startGap = time;
445  endGap = time;
446  return false;
447  }
448 
449  return true;
450  }
