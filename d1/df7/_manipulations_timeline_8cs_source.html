103  {

105 
106  if (zoomFactor <= 0)
107  return;

174  static float PixelToZoom(float x)
175  {
176  const float pixel2Zoom = 1 / 300.0f;
177  x *= pixel2Zoom;
178  if (x < -0.75)
179  {
180  // Rational function that behaves like 1+x on [-0.75,inf) and asymptotically reaches zero on (-inf,-0.75]
181  // The coefficients were obtained by the following constraints:
182  //1) f(-0.75) = 0.25
183  //2) f'(-0.75) = 1 C1 continuity
184  //3) f(-3) = 0.001 (asymptotically zero)
185  return 1 / (98.6667f + 268.444f * x + 189.63f * x * x);
186  }
187  return 1 + x;
188  }
