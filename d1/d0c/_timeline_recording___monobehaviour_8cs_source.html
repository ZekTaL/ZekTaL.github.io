58  internal static UndoPropertyModification[] ProcessMonoBehaviourModification(UndoPropertyModification[] modifications, WindowState state)
59  {
60  if (!state.recording)
61  return modifications;
62 
63  if (state == null || state.editSequence.director == null)
64  return modifications;
65 
66  s_UnprocessedMods.Clear();
67 
68  s_TrackRecorder.PrepareForRecord(state);
69 
70  s_ModsToProcess.Clear();
71  s_ModsToProcess.AddRange(modifications.Reverse());
72 
73  while (s_ModsToProcess.Count > 0)
74  {
75  var modification = s_ModsToProcess[s_ModsToProcess.Count - 1];
76  s_ModsToProcess.RemoveAt(s_ModsToProcess.Count - 1);
77 
78  // grab the clip we need to apply to
79  var modifiedGO = GetGameObjectFromModification(modification);
80  var track = GetTrackForGameObject(modifiedGO, state);
81  if (track != null)
82  {
83  IsRecordingAnimationTrack = true;
84 
85  double startTime = 0;
86  var clip = s_TrackRecorder.PrepareTrack(track, state, modifiedGO, out startTime);
87  if (clip == null)
88  {
89  s_ModsToProcess.Reverse();
90  return s_ModsToProcess.ToArray();
91  }
92  s_RecordState.activeAnimationClip = clip;
93  s_RecordState.activeRootGameObject = state.GetSceneReference(track);
94  s_RecordState.activeGameObject = modifiedGO;
95  s_RecordState.currentFrame = Mathf.RoundToInt((float)startTime);
96 
97  EditorUtility.SetDirty(clip);
98  var toProcess = GatherRelatedModifications(modification, s_ModsToProcess);
99 
100  var animator = s_RecordState.activeRootGameObject.GetComponent<Animator>();
101  var animTrack = track as AnimationTrack;
102 
103  // update preview mode before recording so the correct values get placed (in case we modify offsets)
104  // Case 900624
105  UpdatePreviewMode(toProcess, modifiedGO);
106 
107  // if this is the first position/rotation recording, copy the current position / rotation to the track offset
108  AddTrackOffset(animTrack, toProcess, clip, animator);
109 
110  // same for clip mod clips being created
111  AddClipOffset(animTrack, toProcess, s_TrackRecorder.recordClip, animator);
112 
113  // Check if we need to handle position/rotation offsets
114  var handleOffsets = animator != null && modification.currentValue != null &&
115  modification.currentValue.target == s_RecordState.activeRootGameObject.transform &&
116  HasOffsets(animTrack, s_TrackRecorder.recordClip);
117  if (handleOffsets)
118  {
119  toProcess = HandleEulerModifications(animTrack, s_TrackRecorder.recordClip, clip, s_RecordState.currentFrame * clip.frameRate, toProcess);
120  RemoveOffsets(modification, animTrack, s_TrackRecorder.recordClip, toProcess);
121  }
122 
123  var remaining = AnimationRecording.Process(s_RecordState, toProcess);
124  if (remaining != null && remaining.Length != 0)
125  {
126  s_UnprocessedMods.AddRange(remaining);
127  }
128 
129  if (handleOffsets)
130  {
131  ReapplyOffsets(modification, animTrack, s_TrackRecorder.recordClip, toProcess);
132  }
133 
134  s_TrackRecorder.FinializeTrack(track, state);
135 
136  IsRecordingAnimationTrack = false;
137  }
138  else
139  {
140  s_UnprocessedMods.Add(modification);
141  }
142  }
143 
144 
145  s_TrackRecorder.FinalizeRecording(state);
146 
147  return s_UnprocessedMods.ToArray();
148  }

374  internal static bool HasOffsets(AnimationTrack track, TimelineClip clip)
375  {
376  if (track == null)
377  return false;
378 
379  bool hasClipOffsets = false;
380  bool hasTrackOffsets = false;
381

383  if (clipWrapper != null)
384  hasClipOffsets |= clipWrapper.position != Vector3.zero || clipWrapper.rotation != Quaternion.identity;
385

387  {
388  hasTrackOffsets = track.sceneOffsetPosition != Vector3.zero || track.sceneOffsetRotation != Vector3.zero;

390  else
391  {
392  hasTrackOffsets = (track.position != Vector3.zero || track.rotation != Quaternion.identity);
393  if (!track.inClipMode)
394  hasClipOffsets |= track.infiniteClipOffsetPosition != Vector3.zero || track.infiniteClipOffsetRotation != Quaternion.identity;
395  }
396

457  static UndoPropertyModification[] GatherRelatedModifications(UndoPropertyModification toMatch, List<UndoPropertyModification> list)
458  {
459  var matching = new List<UndoPropertyModification> {toMatch};
460 
461  for (var i = list.Count - 1; i >= 0; i--)
462  {
463  var undo = list[i];
464  if (undo.previousValue.target == toMatch.previousValue.target &&
465  DoesPropertyPathMatch(undo.previousValue.propertyPath, toMatch.previousValue.propertyPath))

467  matching.Add(undo);
468  list.RemoveAt(i);
469  }
470  }

663  internal static UndoPropertyModification[] HandleEulerModifications(AnimationTrack track, TimelineClip clip, AnimationClip animClip, float time, UndoPropertyModification[] mods)
664  {
665  if (mods.Any(x => x.currentValue.propertyPath.StartsWith(kLocalEulerHint) || x.currentValue.propertyPath.StartsWith(kLocalRotation)))
666  {
667  // if there is a rotational offsets, we need to strip the euler hints, since they are used by the animation recording system
668  // over the quaternion.
669  var localToTrack = GetLocalToTrack(track, clip);
670  if (localToTrack.rotation != Quaternion.identity)
671  {
672  if (s_LastTrackWarning != track)
673  {
674  s_LastTrackWarning = track;
675  Debug.LogWarning(kRotationWarning);
676  }
677 
678  Transform transform = mods[0].currentValue.target as Transform;
679  if (transform != null)
680  {
681  var trackToLocal = TimelineAnimationUtilities.RigidTransform.Inverse(localToTrack);
682  // since the euler angles are going to be transformed, we do a best guess at a euler that gives the shortest path
683  var quatMods = mods.Where(x => !x.currentValue.propertyPath.StartsWith(kLocalEulerHint));
684  var eulerMods = FindBestEulerHint(trackToLocal.rotation * transform.localRotation, animClip, time, transform);
685  return quatMods.Union(eulerMods).ToArray();
686  }
687  return mods.Where(x => !x.currentValue.propertyPath.StartsWith(kLocalEulerHint)).ToArray();
688  }
689  }
690  return mods;

693  internal static IEnumerable<UndoPropertyModification> FindBestEulerHint(Quaternion rotation, AnimationClip clip, float time, Transform transform)
694  {
