148  }

154  if (sourceFontAsset.characterLookupTable.TryGetValue(unicode, out character))

164  if (character == null && includeFallbacks && sourceFontAsset.fallbackFontAssetTable != null)

176

181

183  if (k_SearchedAssets.Add(id) == false)
184  continue;
185 
186  // Add reference to this search query
187  sourceFontAsset.FallbackSearchQueryLookup.Add(id);
188 
189  character = GetCharacterFromFontAsset_Internal(unicode, temp, true, fontStyle, fontWeight, out isAlternativeTypeface);
190 
191  if (character != null)
