560  internal override Playable OnCreateClipPlayableGraph(PlayableGraph graph, GameObject go, IntervalTree<RuntimeElement> tree)
561  {
562  if (isSubTrack)
563  throw new InvalidOperationException("Nested animation tracks should never be asked to create a graph directly");
564 
565  List<AnimationTrack> flattenTracks = new List<AnimationTrack>();
566  if (CanCompileClips())
567  flattenTracks.Add(this);
568 
569  var genericRoot = GetGenericRootNode(go);
570  var animatesRootTransformNoMask = AnimatesRootTransform();
571  var animatesRootTransform = animatesRootTransformNoMask && !IsRootTransformDisabledByMask(go, genericRoot);
572  foreach (var subTrack in GetChildTracks())
573  {
574  var child = subTrack as AnimationTrack;
575  if (child != null && child.CanCompileClips())
576  {
577  var childAnimatesRoot = child.AnimatesRootTransform();
578  animatesRootTransformNoMask |= child.AnimatesRootTransform();
579  animatesRootTransform |= (childAnimatesRoot && !child.IsRootTransformDisabledByMask(go, genericRoot));
580  flattenTracks.Add(child);
581  }
582  }
583 
584  // figure out which mode to apply
585  AppliedOffsetMode mode = GetOffsetMode(go, animatesRootTransform);
586  int defaultBlendCount = GetDefaultBlendCount();
587  var layerMixer = CreateGroupMixer(graph, go, flattenTracks.Count + defaultBlendCount);
588  for (int c = 0; c < flattenTracks.Count; c++)
589  {
590  int blendIndex = c + defaultBlendCount;
591  // if the child is masking the root transform, compile it as if we are non-root mode
592  var childMode = mode;
593  if (mode != AppliedOffsetMode.NoRootTransform && flattenTracks[c].IsRootTransformDisabledByMask(go, genericRoot))
594  childMode = AppliedOffsetMode.NoRootTransform;
595 
596  var compiledTrackPlayable = flattenTracks[c].inClipMode ?
597  CompileTrackPlayable(graph, flattenTracks[c], go, tree, childMode) :
598  flattenTracks[c].CreateInfiniteTrackPlayable(graph, go, tree, childMode);
599  graph.Connect(compiledTrackPlayable, 0, layerMixer, blendIndex);
600  layerMixer.SetInputWeight(blendIndex, flattenTracks[c].inClipMode ? 0 : 1);
601  if (flattenTracks[c].applyAvatarMask && flattenTracks[c].avatarMask != null)
602  {
603  layerMixer.SetLayerMaskFromAvatarMask((uint)blendIndex, flattenTracks[c].avatarMask);
604  }
605  }
606 
607  var requiresMotionXPlayable = RequiresMotionXPlayable(mode, go);
608 
609  // In the editor, we may require the motion X playable if we are animating the root transform but it is masked out, because the default poses
610  // need to properly update root motion
611  requiresMotionXPlayable |= (defaultBlendCount > 0 && RequiresMotionXPlayable(GetOffsetMode(go, animatesRootTransformNoMask), go));
612 
613  // Attach the default poses
614  AttachDefaultBlend(graph, layerMixer, requiresMotionXPlayable);
615 
616  // motionX playable not required in scene offset mode, or root transform mode
617  Playable mixer = layerMixer;
618  if (requiresMotionXPlayable)
619  {
620  // If we are animating a root transform, add the motionX to delta playable as the root node
621  var motionXToDelta = AnimationMotionXToDeltaPlayable.Create(graph);
622  graph.Connect(mixer, 0, motionXToDelta, 0);
623  motionXToDelta.SetInputWeight(0, 1.0f);
624  motionXToDelta.SetAbsoluteMotion(UsesAbsoluteMotion(mode));
625  mixer = (Playable)motionXToDelta;
626  }
627 
628 
629 #if UNITY_EDITOR
630  if (!Application.isPlaying)
631  {
632  var animator = GetBinding(go != null ? go.GetComponent<PlayableDirector>() : null);
633  if (animator != null)
634  {
635  GameObject targetGO = animator.gameObject;
636  IAnimationWindowPreview[] previewComponents = targetGO.GetComponents<IAnimationWindowPreview>();
637 
638  m_HasPreviewComponents = previewComponents.Length > 0;
639  if (m_HasPreviewComponents)
640  {
641  foreach (var component in previewComponents)
642  {
643  mixer = component.BuildPreviewGraph(graph, mixer);
644  }
645  }
646  }
647  }
648 #endif
649 
650  return mixer;
651  }
