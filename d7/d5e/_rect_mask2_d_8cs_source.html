202  }

204  //TODO See if an IsActive() test would work well here or whether it might cause unexpected side effects (re case 776771)

206  // if the parents are changed
207  // or something similar we
208  // do a recalculate here
209  if (m_ShouldRecalculateClipRects)
210  {
211  MaskUtilities.GetRectMasksForClip(this, m_Clippers);
212  m_ShouldRecalculateClipRects = false;
213  }
214 
215  // get the compound rects from
216  // the clippers that are valid
217  bool validRect = true;
218  Rect clipRect = Clipping.FindCullAndClipWorldRect(m_Clippers, out validRect);
219 
220  // If the mask is in ScreenSpaceOverlay/Camera render mode, its content is only rendered when its rect
221  // overlaps that of the root canvas.
222  RenderMode renderMode = Canvas.rootCanvas.renderMode;
223  bool maskIsCulled =
224  (renderMode == RenderMode.ScreenSpaceCamera || renderMode == RenderMode.ScreenSpaceOverlay) &&
225  !clipRect.Overlaps(rootCanvasRect, true);
226 
227  if (maskIsCulled)
228  {
229  // Children are only displayed when inside the mask. If the mask is culled, then the children
230  // inside the mask are also culled. In that situation, we pass an invalid rect to allow callees
231  // to avoid some processing.
232  clipRect = Rect.zero;
233  validRect = false;
234  }
235 
236  if (clipRect != m_LastClipRectCanvasSpace)
237  {
238  foreach (IClippable clipTarget in m_ClipTargets)
239  {
240  clipTarget.SetClipRect(clipRect, validRect);
241  }
242 
243  foreach (MaskableGraphic maskableTarget in m_MaskableTargets)
244  {
245  maskableTarget.SetClipRect(clipRect, validRect);
246  maskableTarget.Cull(clipRect, validRect);
247  }
248  }
