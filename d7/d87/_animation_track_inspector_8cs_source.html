383  void DoOffsetManipulator()
384  {
385  if (targets.Length > 1) //do not edit the track offset on a multiple selection
386  return;
387 
388  if (timelineWindow == null || timelineWindow.state == null || timelineWindow.state.editSequence.director == null)
389  return;
390 
391  AnimationTrack animationTrack = target as AnimationTrack;
392  if (animationTrack != null && (animationTrack.trackOffset == TrackOffset.ApplyTransformOffsets) && m_OffsetEditMode != TimelineAnimationUtilities.OffsetEditMode.None)
393  {
394  var boundObject = TimelineUtility.GetSceneGameObject(timelineWindow.state.editSequence.director, animationTrack);
395  var boundObjectTransform = boundObject != null ? boundObject.transform : null;
396 
397  var offsets = TimelineAnimationUtilities.GetTrackOffsets(animationTrack, boundObjectTransform);
398  EditorGUI.BeginChangeCheck();
399 
400  switch (m_OffsetEditMode)
401  {
402  case TimelineAnimationUtilities.OffsetEditMode.Translation:
403  offsets.position = Handles.PositionHandle(offsets.position, (Tools.pivotRotation == PivotRotation.Global)
404  ? Quaternion.identity
405  : offsets.rotation);
406  break;
407  case TimelineAnimationUtilities.OffsetEditMode.Rotation:
408  offsets.rotation = Handles.RotationHandle(offsets.rotation, offsets.position);
409  break;
410  }
411 
412  if (EditorGUI.EndChangeCheck())
413  {
414  UndoExtensions.RegisterTrack(animationTrack, "Inspector");
415  TimelineAnimationUtilities.UpdateTrackOffset(animationTrack, boundObjectTransform, offsets);
416  Evaluate();
417  Repaint();
418  }
419  }
