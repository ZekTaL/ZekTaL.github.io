247  private bool IsProjectGeneratedFor(string path, out ProjectGenerationFlag missingFlag)
248  {
249  missingFlag = ProjectGenerationFlag.None;
250 
251  // No need to check when opening the whole solution
252  if (string.IsNullOrEmpty(path))
253  return true;
254 
255  // We only want to check for cs scripts
256  if (ProjectGeneration.ScriptingLanguageFor(path) != ScriptingLanguage.CSharp)
257  return true;
258 
259  // Even on windows, the package manager requires relative path + unix style separators for queries
260  var basePath = _generator.ProjectDirectory;
261  var relativePath = FileUtility
262  .NormalizeWindowsToUnix(path)
263  .Replace(basePath, string.Empty)
264  .Trim(FileUtility.UnixSeparator);
265 
266  var packageInfo = UnityEditor.PackageManager.PackageInfo.FindForAssetPath(relativePath);
267  if (packageInfo == null)
268  return true;
269 
270  var source = packageInfo.source;
271  if (!Enum.TryParse<ProjectGenerationFlag>(source.ToString(), out var flag))
272  return true;
273 
274  if (_generator.AssemblyNameProvider.ProjectGenerationFlag.HasFlag(flag))
275  return true;
276 
277  // Return false if we found a source not flagged for generation
278  missingFlag = flag;
279  return false;
280  }

282  private bool OpenWindowsApp(string path, int line)
283  {
284  var progpath = FileUtility
285  .FindPackageAssetFullPath("COMIntegration a:packages", "COMIntegration.exe")
286  .FirstOrDefault();
287 
288  if (string.IsNullOrWhiteSpace(progpath))
289  return false;
290 
291  string absolutePath = "";
292  if (!string.IsNullOrWhiteSpace(path))
293  {
294  absolutePath = Path.GetFullPath(path);
295  }
296 
297  // We remove all invalid chars from the solution filename, but we cannot prevent the user from using a specific path for the Unity project
298  // So process the fullpath to make it compatible with VS
299  var solution = GetOrGenerateSolutionFile(path);
300  if (!string.IsNullOrWhiteSpace(solution))
301  {
302  solution = $"\"{solution}\"";
303  solution = solution.Replace("^", "^^");
304  }
305 
306  var process = new Process
307  {
308  StartInfo = new ProcessStartInfo
309  {
310  FileName = progpath,
311  Arguments = $"\"{CodeEditor.CurrentEditorInstallation}\" {solution} \"{absolutePath}\" {line}",
312  CreateNoWindow = true,
313  UseShellExecute = false,
314  RedirectStandardOutput = true,
315  RedirectStandardError = true,
316  }
317  };
318  var result = process.Start();
319 
320  while (!process.StandardOutput.EndOfStream)
321  {
322  var outputLine = process.StandardOutput.ReadLine();
323  if (outputLine == "displayProgressBar")
324  {
325  EditorUtility.DisplayProgressBar("Opening Visual Studio", "Starting up Visual Studio, this might take some time.", .5f);
326  }
327 
328  if (outputLine == "clearprogressbar")
329  {
330  EditorUtility.ClearProgressBar();
331  }
332  }
333 
334  var errorOutput = process.StandardError.ReadToEnd();
335  if (!string.IsNullOrEmpty(errorOutput))
336  {
337  Console.WriteLine("Error: \n" + errorOutput);
338  }
339 
340  process.WaitForExit();
341  return result;
342  }
