342  public static string Elipsify(string label, float destinationWidth, float neededWidth)
343  {
344  var ret = label;
345 
346  if (label.Length == 0)
347  return ret;
348 
349  if (destinationWidth < neededWidth)
350  {
351  float averageWidthOfOneChar = neededWidth / label.Length;
352  int floor = Mathf.Max((int)Mathf.Floor(destinationWidth / averageWidthOfOneChar), 0);
353 
354  if (floor < k_Elipsis.Length)
355  ret = string.Empty;
356  else if (floor == k_Elipsis.Length)
357  ret = k_Elipsis;
358  else if (floor < label.Length)
359  ret = label.Substring(0, floor - k_Elipsis.Length) + k_Elipsis;
360  }
361 
362  return ret;
363  }
