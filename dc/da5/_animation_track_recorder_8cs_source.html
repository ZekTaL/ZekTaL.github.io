30  {

32 
33  // if we are not in clip mode, we simply use the track clip
34  var animationTrack = (AnimationTrack)track;
35 
36  // ignore recording if we are in Legacy auto mode
37  startTime = -1;
38  var parentTrack = TimelineUtility.GetSceneReferenceTrack(track) as AnimationTrack;
39  if (parentTrack != null && parentTrack.trackOffset == TrackOffset.Auto)
40  return null;
41 
42  if (!animationTrack.inClipMode)
43  {
44  var trackClip = animationTrack.GetOrCreateClip();
45  startTime = trackClip.frameRate * state.editSequence.time;
46 
47  // Make the first key be at time 0 of the clip
48  if (trackClip.empty)
49  {
50  animationTrack.infiniteClipTimeOffset = 0; // state.time;
51  animationTrack.infiniteClipPreExtrapolation = TimelineClip.ClipExtrapolation.Hold;
52  animationTrack.infiniteClipPostExtrapolation = TimelineClip.ClipExtrapolation.Hold;
53  }
54 
55  animationClip = trackClip;
56  }
57  else
58  {
59  TimelineClip activeClip = null;
60 
61  // if it fails, but returns no clip, we can add one.
62  if (!track.FindRecordingClipAtTime(state.editSequence.time, out activeClip) && activeClip != null)
63  {
64  return null;
65  }
66 
67  if (activeClip == null)
68  {
69  activeClip = AddRecordableClip(track, state, state.editSequence.time);
70  }
71 
72  var clip = activeClip.animationClip;
73 
74  // flags this as the clip being recorded for the track
75  var clipTime = state.editSequence.time - activeClip.start;
76 
77  // if we are in the past
78  if (clipTime < 0)
79  {
80  Undo.RegisterCompleteObjectUndo(clip, "Record Key");
81  UndoExtensions.RegisterTrack(track, "Prepend Key");
82  ShiftAnimationClip(clip, (float)-clipTime);
83  activeClip.start = state.editSequence.time;
84  activeClip.duration += -clipTime;
85  clipTime = 0;
86  }
87 
88  m_ClipTime = clipTime;
89  recordClip = activeClip;
90  startTime = recordClip.ToLocalTimeUnbound(state.editSequence.time) * clip.frameRate;
91  m_needRebuildRects = clip.empty;
92 
93  animationClip = clip;
94  }
95

97  m_CurveCount = GetCurveCount(animationClip);
98  m_TrackHasPreviewComponents = animationTrack.hasPreviewComponents;
99 
100  return animationClip;
