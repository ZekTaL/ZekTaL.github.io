720  {
721  if (!IsActive() || !IsInteractable() || m_Dropdown != null)
722  return;
723 
724  // Get root Canvas.
725  var list = ListPool<Canvas>.Get();
726  gameObject.GetComponentsInParent(false, list);
727  if (list.Count == 0)
728  return;
729 
730  // case 1064466 rootCanvas should be last element returned by GetComponentsInParent()
731  var listCount = list.Count;
732  Canvas rootCanvas = list[listCount - 1];
733  for (int i = 0; i < listCount; i++)
734  {
735  if (list[i].isRootCanvas || list[i].overrideSorting)
736  {
737  rootCanvas = list[i];
738  break;
739  }
740  }
741 
742  ListPool<Canvas>.Release(list);
743 
744  if (!validTemplate)
745  {
746  SetupTemplate();
747  if (!validTemplate)
748  return;
749  }

751  m_Template.gameObject.SetActive(true);
752 
753  // popupCanvas used to assume the root canvas had the default sorting Layer, next line fixes (case 958281 - [UI] Dropdown list does not copy the parent canvas layer when the panel is opened)
754  m_Template.GetComponent<Canvas>().sortingLayerID = rootCanvas.sortingLayerID;
755 
756  // Instantiate the drop-down template
757  m_Dropdown = CreateDropdownList(m_Template.gameObject);
758  m_Dropdown.name = "Dropdown List";
759  m_Dropdown.SetActive(true);
760 
761  // Make drop-down RectTransform have same values as original.
762  RectTransform dropdownRectTransform = m_Dropdown.transform as RectTransform;
763  dropdownRectTransform.SetParent(m_Template.transform.parent, false);
764 
765  // Instantiate the drop-down list items
766 
767  // Find the dropdown item and disable it.
768  DropdownItem itemTemplate = m_Dropdown.GetComponentInChildren<DropdownItem>();
769 
770  GameObject content = itemTemplate.rectTransform.parent.gameObject;
771  RectTransform contentRectTransform = content.transform as RectTransform;
772  itemTemplate.rectTransform.gameObject.SetActive(true);
773 
774  // Get the rects of the dropdown and item
775  Rect dropdownContentRect = contentRectTransform.rect;
776  Rect itemTemplateRect = itemTemplate.rectTransform.rect;
777 
778  // Calculate the visual offset between the item's edges and the background's edges
779  Vector2 offsetMin = itemTemplateRect.min - dropdownContentRect.min + (Vector2)itemTemplate.rectTransform.localPosition;
780  Vector2 offsetMax = itemTemplateRect.max - dropdownContentRect.max + (Vector2)itemTemplate.rectTransform.localPosition;
781  Vector2 itemSize = itemTemplateRect.size;
782 
783  m_Items.Clear();
784 
785  Toggle prev = null;
786  var optionsCount = options.Count;
787  for (int i = 0; i < optionsCount; ++i)
788  {
789  OptionData data = options[i];
790  DropdownItem item = AddItem(data, value == i, itemTemplate, m_Items);
791  if (item == null)
792  continue;
793 
794  // Automatically set up a toggle state change listener
795  item.toggle.isOn = value == i;
796  item.toggle.onValueChanged.AddListener(x => OnSelectItem(item.toggle));
797 
798  // Select current option
799  if (item.toggle.isOn)
800  item.toggle.Select();
801 
802  // Automatically set up explicit navigation
803  if (prev != null)
804  {
805  Navigation prevNav = prev.navigation;
806  Navigation toggleNav = item.toggle.navigation;
807  prevNav.mode = Navigation.Mode.Explicit;
808  toggleNav.mode = Navigation.Mode.Explicit;
809

811  prevNav.selectOnRight = item.toggle;
812  toggleNav.selectOnLeft = prev;
813  toggleNav.selectOnUp = prev;
814 
815  prev.navigation = prevNav;
816  item.toggle.navigation = toggleNav;
817  }

819  }

822  Vector2 sizeDelta = contentRectTransform.sizeDelta;
823  sizeDelta.y = itemSize.y * m_Items.Count + offsetMin.y - offsetMax.y;
824  contentRectTransform.sizeDelta = sizeDelta;
825

827  if (extraSpace > 0)
828  dropdownRectTransform.sizeDelta = new Vector2(dropdownRectTransform.sizeDelta.x, dropdownRectTransform.sizeDelta.y - extraSpace);
829 
830  // Invert anchoring and position if dropdown is partially or fully outside of canvas rect.
831  // Typically this will have the effect of placing the dropdown above the button instead of below,
832  // but it works as inversion regardless of initial setup.
833  Vector3[] corners = new Vector3[4];
834  dropdownRectTransform.GetWorldCorners(corners);
835 
836  RectTransform rootCanvasRectTransform = rootCanvas.transform as RectTransform;
837  Rect rootCanvasRect = rootCanvasRectTransform.rect;
838  for (int axis = 0; axis < 2; axis++)
839  {
840  bool outside = false;
841  for (int i = 0; i < 4; i++)
842  {
843  Vector3 corner = rootCanvasRectTransform.InverseTransformPoint(corners[i]);
844  if ((corner[axis] < rootCanvasRect.min[axis] && !Mathf.Approximately(corner[axis], rootCanvasRect.min[axis])) ||
845  (corner[axis] > rootCanvasRect.max[axis] && !Mathf.Approximately(corner[axis], rootCanvasRect.max[axis])))
846  {
847  outside = true;
848  break;
849  }
850  }
851  if (outside)
852  RectTransformUtility.FlipLayoutOnAxis(dropdownRectTransform, axis, false, false);
853  }
854

856  for (int i = 0; i < itemsCount; i++)
857  {

859  itemRect.anchorMin = new Vector2(itemRect.anchorMin.x, 0);
