205  return;
206  }
207  }

210  if (currentPointerData.pointerEnter == newEnterTarget && newEnterTarget)
211  return;
212 
213  GameObject commonRoot = FindCommonRoot(currentPointerData.pointerEnter, newEnterTarget);
214 
215  // and we already an entered object from last time
216  if (currentPointerData.pointerEnter != null)
217  {
218  // send exit handler call to all elements in the chain
219  // until we reach the new target, or null!
220  Transform t = currentPointerData.pointerEnter.transform;
221 
222  while (t != null)
223  {
224  // if we reach the common root break out!
225  if (commonRoot != null && commonRoot.transform == t)
226  break;
227 
228  ExecuteEvents.Execute(t.gameObject, currentPointerData, ExecuteEvents.pointerExitHandler);
229  currentPointerData.hovered.Remove(t.gameObject);
230  t = t.parent;
231  }
232  }

235  currentPointerData.pointerEnter = newEnterTarget;
