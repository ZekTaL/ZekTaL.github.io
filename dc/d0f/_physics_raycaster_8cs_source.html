100  {
101  if (eventCamera == null)

103 
104  var eventPosition = Display.RelativeMouseAt(eventData.position);
105  if (eventPosition != Vector3.zero)
106  {
107  // We support multiple display and display identification based on event position.
108  eventDisplayIndex = (int)eventPosition.z;
109 
110  // Discard events that are not part of this display so the user does not interact with multiple displays at once.
111  if (eventDisplayIndex != eventCamera.targetDisplay)
112  return false;
113  }
114  else
115  {
116  // The multiple display system is not supported on all platforms, when it is not supported the returned position
117  // will be all zeros so when the returned index is 0 we will default to the event data to be safe.
118  eventPosition = eventData.position;
119  }
120 
121  // Cull ray casts that are outside of the view rect. (case 636595)
122  if (!eventCamera.pixelRect.Contains(eventPosition))
123  return false;
124

126  // compensate far plane distance - see MouseEvents.cs
127  float projectionDirection = ray.direction.z;
128  distanceToClipPlane = Mathf.Approximately(0.0f, projectionDirection)
129  ? Mathf.Infinity
130  : Mathf.Abs((eventCamera.farClipPlane - eventCamera.nearClipPlane) / projectionDirection);
131  return true;
