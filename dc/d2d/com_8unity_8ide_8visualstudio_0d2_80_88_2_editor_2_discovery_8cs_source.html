40  private static bool IsCandidateToDiscovery(string path)
41  {
42  if (File.Exists(path) && VisualStudioEditor.IsWindows && Regex.IsMatch(path, "devenv.exe$", RegexOptions.IgnoreCase))
43  return true;
44 
45  if (Directory.Exists(path) && VisualStudioEditor.IsOSX && Regex.IsMatch(path, "Visual\\s?Studio(?!.*Code.*).*.app$", RegexOptions.IgnoreCase))

51  public static bool TryDiscoverInstallation(string editorPath, out IVisualStudioInstallation installation)
52  {
53  installation = null;
54 
55  if (string.IsNullOrEmpty(editorPath))
56  return false;
57 
58  if (!IsCandidateToDiscovery(editorPath))
59  return false;
60 
61  // On windows we use the executable directly, so we can query extra information
62  var fvi = editorPath;
63 
64  // On Mac we use the .app folder, so we need to access to main assembly
65  if (VisualStudioEditor.IsOSX)
66  fvi = Path.Combine(editorPath, "Contents", "Resources", "lib", "monodevelop", "bin", "VisualStudio.exe");
67 
68  if (!File.Exists(fvi))
69  return false;
70 
71  // VS preview are not using the isPrerelease flag so far
72  // On Windows FileDescription contains "Preview", but not on Mac
73  var vi = FileVersionInfo.GetVersionInfo(fvi);
74  var version = new Version(vi.ProductVersion);
75  var isPrerelease = vi.IsPreRelease || string.Concat(editorPath, "/" + vi.FileDescription).ToLower().Contains("preview");
76 
77  installation = new VisualStudioInstallation()
78  {
79  IsPrerelease = isPrerelease,
80  Name = $"{vi.FileDescription}{(isPrerelease && VisualStudioEditor.IsOSX ? " Preview" : string.Empty)} [{version.ToString(3)}]",
81  Path = editorPath,
82  Version = version
83  };
84  return true;
85  }

132  private static IEnumerable<VisualStudioInstallation> QueryVsWhere()
133  {
134  var progpath = FileUtility
135  .FindPackageAssetFullPath("VSWhere a:packages", "vswhere.exe")
136  .FirstOrDefault();
137 
138  if (string.IsNullOrWhiteSpace(progpath))
139  return Enumerable.Empty<VisualStudioInstallation>();
140 
141  var process = new Process
142  {
143  StartInfo = new ProcessStartInfo
144  {
145  FileName = progpath,
146  Arguments = "-prerelease -format json -utf8",
147  UseShellExecute = false,
148  CreateNoWindow = true,
149  RedirectStandardOutput = true,
150  RedirectStandardError = true,
151  }
152  };
153 
154  using (process)
155  {
156  var json = string.Empty;
157 
158  process.OutputDataReceived += (o, e) => json += e.Data;
159  process.Start();
160  process.BeginOutputReadLine();
161  process.WaitForExit();
162 
163  var result = VsWhereResult.FromJson(json);
164  return result.ToVisualStudioInstallations();
165  }
166  }
